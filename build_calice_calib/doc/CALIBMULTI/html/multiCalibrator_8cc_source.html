255  }

281 
282  // get a working copy of the raw histogram

284

288  const Double_t minRange = fitHist->GetBinLowEdge(1);
289  const Double_t maxRange = fitHist->GetBinLowEdge(nBins) + fitHist->GetBinWidth(nBins);
290  const Double_t binWidth = fitHist->GetBinWidth(1);
291  const Double_t entries = fitHist->GetEntries();
292 
293  streamlog_out(MESSAGE2) << "Rebinned histogram has " << nBins << " bins in range from " << minRange << " to " << maxRange << std::endl;
294  streamlog_out(MESSAGE2) << "BinWidth: " << binWidth << std::endl;
295 
296 
297  // as the convoluted function has no start value apprixmation, we have to do it by hand
298  now = std::time(NULL);
299  streamlog_out(MESSAGE1) << "estimating start values for convoluted fit " << std::asctime(std::localtime(&now));
300 
301  g = new TF1("g","gaus",minRange,maxRange);
302  l = new TF1("l","landau",minRange,maxRange);
303 
304  fitHist->Fit(l,"RQ0N");
305  _pedHist->Fit(g,"RQ0N");
306 
307  Double_t pedestal = g->GetParameter(1);
308 
309  /* generate the functions for the convolution, we need the properly normalized versions
310  also one function should have the mean/MPV fixed at the origin, for our case the gaussian */
311  gg = new TF1("gg","TMath::Gaus(x,0,[0],1)",-1.*_gausRange,_gausRange);
312  gg->SetParName(0,"sigma");
313  ll = new TF1("ll","TMath::Landau(x,[0],[1],1)",minRange,maxRange);
314  ll->SetParNames("MPV","sigma");
315 
316  /* initialize the functions for the convolution
317  first approximation:
318  gaus broadening is pure electronic noise (pedestal fit width)
319  landau parameters are taken from landau fit of the signal */
320  gg->SetParameter(0,g->GetParameter(2)); // Gaus sigma
321  ll->SetParameter(0,l->GetParameter(1)); // Landau MPV
322  ll->SetParameter(1,l->GetParameter(2)); // Landau sigma
323 
324  // get convoluted function
325  convoluter = new TConvolution("convolution",gg,ll);
326  TF1* conv = convoluter->GetConvolutedFunction();
327 
328  // normalize
329  conv->SetParameter(3,binWidth*entries);
330 
331  // set fit range
332  Double_t fitRangeMax = l->GetParameter(1)+6*l->GetParameter(2); // TO-DO, fit range should not be hardcoded
333  Double_t fitRangeMin = l->GetParameter(1)-6*l->GetParameter(2);
334  conv->SetRange(fitRangeMin,fitRangeMax);
335 
336  // fitting signal
337  now = std::time(NULL);
338  streamlog_out(MESSAGE1) << "fitting convoluted function " << std::asctime(std::localtime(&now));
339  fitHist->Fit(conv,"Q0RN");
340 
341  Double_t MIPpos = conv->GetMaximumX();
342  Double_t MIP = MIPpos - pedestal;
343  streamlog_out(MESSAGE) << "pedestal position: " << pedestal << " MIP position: " << MIPpos << " MIP: " << MIP << std::endl;
344 
345 
346  now = std::time(NULL);
347  streamlog_out(MESSAGE2) << "generating histogram from fit function " << std::asctime(std::localtime(&now));
348 
349  /* we need something faster than a function generated by convolution for the later processing
350  therefore a histogram is generated from the function
351  also the histogram is readable without the ROOT extension when saved in the file */
352  conv->SetRange(MIPpos-0.5*MIP,maxRange); // extend range
353 
354  fitResultHist = new TH1F("fitResultHist","one MIP histogram",nBins,minRange,maxRange);
355  fitResultHist->Add(conv);
356 
357  /* How good is the fit? How much double events exist?
358  generate residual */
359  now = std::time(NULL);
360  streamlog_out(MESSAGE2) << "generating substracted histogram " << std::asctime(std::localtime(&now)) ;
361 
362  subHist = (TH1*) fitHist->Clone("substractedHist");
363  subHist->GetListOfFunctions()->Clear();
364  subHist->Sumw2();
365  subHist->Add(fitResultHist,-1);
366 
367 
368  /* We need to know the distribution of the two MIP events, for the calculation of the treshold.
369  To save CPU the distribution is simulated from the single MIP distribution. */
370  now = std::time(NULL);
371  streamlog_out(MESSAGE2) << "generating distributions for threshold calculation" << std::asctime(std::localtime(&now)) ;
372 
373  /* Before playing with the parameters of the result: backup
374  using vector, as some compiler options do not allow arrays to be initalized with a const generated within the function */
375  const int NPAR = conv->GetNpar();
376  std::vector<Double_t> parBackup;
377  parBackup.reserve(NPAR);
378  conv->GetParameters(&(parBackup[0]));
379 
380  // generate finer histogram of the fit result (for reference)
381  fitHistFine = new TH1F("fitHistfine","one MIP histogram",nBins*GRANFAC,minRange,maxRange);
382  conv->SetParameter(3,conv->GetParameter(3)/(float)GRANFAC); // renormalize to new binning
383  fitHistFine->Add(conv);
384  conv->SetParameters(&(parBackup[0]));
385 
386  /* We need the MIP distribution without pedestal & electronics noise broadening for the later calculation of the two MIP distribution */
387  pureMIPhist = new TH1F("pureMIPhist","one MIP histogram without pedestal and electronic noise",nBins*GRANFAC,minRange,maxRange);
388  conv->SetParameter(0,sqrt(pow(conv->GetParameter(0),2)-pow(g->GetParameter(2),2))); // quadratically substract the pedestal width (electronics noise)
389  conv->SetParameter(1,conv->GetParameter(1)-pedestal); // shift for pedestal
390  conv->SetRange(MIP-0.5*MIP,maxRange);
391  pureMIPhist->Add(conv);
392  conv->SetParameters(&(parBackup[0]));
393 
394  now = std::time(NULL);
395  streamlog_out(MESSAGE2) << "generating MIP & two MIP histogram " << std::asctime(std::localtime(&now));
396 
397  oneMIPhist = new TH1F("oneMIPhist","one MIP histogram",nBins*GRANFAC,minRange,maxRange);
398  for (unsigned i(0); i<1000000;i++) oneMIPhist->Fill(pureMIPhist->GetRandom() + g->GetRandom());
399 
400  twoMIPhist = new TH1F("twoMIPhist","two MIP histogram",nBins*GRANFAC,minRange,maxRange);
401  for (unsigned i(0); i<1000000;i++) twoMIPhist->Fill(pureMIPhist->GetRandom() + pureMIPhist->GetRandom() + g->GetRandom());
402 
403  // now calculate threshold from distribution
404 
405  std::sort(_purityList.begin(), _purityList.end()); // order of values is important for an efficient threshold calculation;
406 
407  Int_t bin=0;
408  Double_t rejectInt=0;
409  LCCollection *thresholdCollection = new LCCollectionVec(LCIO::LCGENERICOBJECT);
410  LCCollection *lossCollection = new LCCollectionVec(LCIO::LCGENERICOBJECT);
411 
412  LCCollection *referenceLossCollection = 0;
413  if (_referenceThresholdColName!= "") referenceLossCollection = new LCCollectionVec(LCIO::LCGENERICOBJECT);
414 
415  for (int pos=(int)_purityList.size()-1; pos>=0 && bin<twoMIPhist->GetNbinsX();--pos) {
416 
417  float purity = _purityList[pos];
418  float limit = (1. - purity/100.)*twoMIPhist->GetEntries();

420 
421  float thresholdValue = twoMIPhist->GetBinLowEdge(bin-1);
422  float thresholdValueErr = twoMIPhist->GetBinWidth(bin-1)/2.;

431  lossCollection->addElement(loss);
432 
433  }
434 
435  if (referenceLossCollection)
436  for (ThresholdMap_t::const_iterator iter = _referenceThresholds.begin(); iter != _referenceThresholds.end();++iter) {

462  }
463

465

473 
474  streamlog_out(MESSAGE) << " double particle events: " << doubleParticles << std::endl;

477  streamlog_out(MESSAGE) << "done " << std::asctime(std::localtime(&now));
