276  bool MipTrackFinder::doFurtherAnalysis()
277  {
278  /* here apply criteria to use the number of hits in HCAL to apply the muon*/
279  bool analyse = false;
280 
281  if(_useECALandTCMT == true)
282  {
283  streamlog_out(DEBUG4) << "Using ECAL and TCMT information" << endl;
284 
285  /* ECAL information*/
286  std::pair<int, float> nhits_and_esum(extractNumberOfHitsAndEnergySum(_ecalInputCol));
287  int ecalNumberOfHits = nhits_and_esum.first;
288 
289  /* TCMT information*/
290  nhits_and_esum = extractNumberOfHitsAndEnergySum(_tcmtInputCol);
291  float tcmtEnergySum = nhits_and_esum.second;
292 
293  if((ecalNumberOfHits > _ecalMinNoHits) && (ecalNumberOfHits < _ecalMaxNoHits)
294  && (tcmtEnergySum > _tcmtMinEnergySum))
295  {
296  analyse = true;
297  streamlog_out(DEBUG4) << "ECAL and TCMT criteria reached. "
298  << "Going to do further processing." << endl;
299  }
300  else
301  {
302  analyse = false;
303  streamlog_out(DEBUG4) << "ECAL and TCMT criteria *not* reached. "
304  << "Event will be skipped." << endl;
305  }
306  }
307  else
308  {
309  if (_isNonMuonRun == true)
310  {
311  /* this is just for non-muon runs case!*/
312  std::pair<int, float> nhits_and_esum(extractNumberOfHitsAndEnergySum(_ahcalInputCol));
313  int nhits_hcal = nhits_and_esum.first;
314  streamlog_out(DEBUG4) << "hcal hits " << nhits_hcal << endl;
315 
316  if (nhits_hcal < _ahcalMaxNoHits)
317  {
318  analyse = true;
319  }
320  else
321  {
322  analyse = false;
323  }
324  }
325  else
326  {
327  analyse = true;
328  }
329  streamlog_out(DEBUG4) << " ->AHCAL only analysis will start." << endl;
330  }
331 
332  return analyse;
333  }

361  LCTypedVector<CalorimeterHit>* MipTrackFinder::extractAHCALInformation()
362  {
363  streamlog_out(DEBUG4)<<"\n extractAHCALInformation:"<<endl;
364  /*check if there is a list of bad modules*/
365  bool haveListOfBadModules = parameterSet("AHCALBadModules");
366 
367  streamlog_out(DEBUG4)<<" haveListOfBadModules: "<<haveListOfBadModules<<endl;
368 
369  LCTypedVector<CalorimeterHit>* hcalVec = new LCTypedVector<CalorimeterHit> (_ahcalInputCol);
370 
371 
372  UTIL::LCRelationNavigator *navigator = NULL;
373  if (_ahcalAmpInputCol != NULL)
374  {
375  navigator = new UTIL::LCRelationNavigator(_ahcalAmpInputCol);
376  }
377  else {
378  std::cout<<" no relation collection, aborting"<<endl;
379  exit(1);
380  }
381 
382  _allAhcalHitsInEventVec = *hcalVec;
383 
384 
385 
386  /*counter for debug purposes*/
387  int counter = 0;
388 
389  for (unsigned int i = 0; i < hcalVec->size(); ++i)
390  {
391  const int cellID = (*hcalVec)[i]->getCellID0();
392  const int module = _mapper->getModuleFromCellID(cellID);
393 
394  bool hitIsInBadModule = false;
395 
396  if (haveListOfBadModules == true) {
397  if (std::find(_badAhcalModulesVec.begin(), _badAhcalModulesVec.end(), module) != _badAhcalModulesVec.end()){
398  hitIsInBadModule = true;
399  }
400 
401  if ( hitIsInBadModule ){
402  streamlog_out(DEBUG4)<<" -> "<<counter<<" hit in module "<< module << "-> on list of bad modules" <<endl;
403  }
404  else{
405  streamlog_out(DEBUG4)<<" -> "<<counter<<" hit in module "<< module << "-> NOT on list of bad modules" <<endl;
406  }
407  }
408 
409 
410  // get the amplitude without temperature correction
411 
412  const LCObjectVec &amplVec = navigator->getRelatedToObjects((*hcalVec)[i]);
413 
414  float ampMipNoTempNoSat;
415  ampMipNoTempNoSat=0.;
416  if (amplVec.size() > 0)
417  {
418  LCGenericObject *obj = dynamic_cast<LCGenericObject*>(amplVec[0]);
419  AhcAmplitude *ahcAmpl = new AhcAmplitude(obj);
420  ampMipNoTempNoSat=ahcAmpl->getAmplNOTTemperatureCorrMIP();
421  delete ahcAmpl;
422  }
423 
424  // get the amplitude after temperature correction & saturated corrected
425 
426  float ampMipFinal;
427  ampMipFinal=(*hcalVec)[i]->getEnergy();
428 
429  // choose amplitude to be used
430 
431  float mipCut;
432  mipCut=0;
433  if (_mipCutVariable==1){
434  mipCut=ampMipFinal;
435  }
436  else if (_mipCutVariable==0){
437  mipCut=ampMipNoTempNoSat;
438  }
439 
440  // Now do the selection(!)
441 
442  if (
443  (haveListOfBadModules == true && hitIsInBadModule == true)
444  || (mipCut < _mipCutValue) )
445  {
446  counter++;
447  hcalVec->erase(hcalVec->begin() + i); /*remove this hit from the vector*/
448  i--; /* go back one step not to miss the next hit*/
449  }
450 
451  }
452 
453  streamlog_out(DEBUG4)<<" -> "<<counter<<" hits do not fullfill MIP cut and/or are in bad modules"<<endl;
454  streamlog_out(DEBUG4)<<" -> selected AHCAL hits: "<<hcalVec->size()<<endl;
455 
456  delete navigator;
457 
458  return hcalVec;
459  }

465  EVENT::ClusterVec MipTrackFinder::findTrackClusters(LCTypedVector<CalorimeterHit>* hcalVec)
466  {
467  streamlog_out(DEBUG4)<<"\n--------------- Start to find track clusters"<<endl;
468 
469  EVENT::ClusterVec clusterVec;
470 
471  LCTypedVector<CalorimeterHit>::iterator seedIter = hcalVec->begin();
472  while(seedIter != hcalVec->end())
473  {
474  CalorimeterHit *seedHit = (*seedIter);
475  const int cellID = seedHit->getCellID0();
476  /*erase seed hit*/
477  hcalVec->erase(seedIter);
478 
479  /*use only hits in fine modules to find tracks*/
480  const int K = _mapper->getDecoder()->getKFromCellID(cellID);
481  if (K <= 30)
482  {
483  ClusterImpl *cluster = new ClusterImpl;
484  /*first member of the cluster is the seed hit*/
485  cluster->addHit(seedHit, 1.);
486 
487 
488  LCTypedVector<CalorimeterHit>::iterator potentialTrackHitIter = hcalVec->begin();
489  while(potentialTrackHitIter != hcalVec->end())
490  {
491  float differenceInX = fabs((*potentialTrackHitIter)->getPosition()[0] - seedHit->getPosition()[0]);
492  float differenceInY = fabs((*potentialTrackHitIter)->getPosition()[1] - seedHit->getPosition()[1]);
493 
494  CellDescription* seedHitCellDescription = _cellDescriptions->getByCellID(cellID);
495  const float seedHitTileSize = seedHitCellDescription->getSizeX();
496  const float potentialTrackHitTileSize =
497  _cellDescriptions->getByCellID((*potentialTrackHitIter)->getCellID0())->getSizeX();
498 
499  float meanCellSize = 0.5 * (potentialTrackHitTileSize + seedHitTileSize);
500 
501  if (differenceInX < meanCellSize && differenceInY < meanCellSize)
502  {
503  streamlog_out(DEBUG4)<<" New track hit found "<<endl;
504 
505  /*add hit to the track cluster*/
506  cluster->addHit((*potentialTrackHitIter), 1.);
507 
508  /*erase track hit from the HCAL seeds vector*/
509  potentialTrackHitIter = hcalVec->erase(potentialTrackHitIter);
510  }
511  else
512  {
513  ++potentialTrackHitIter;
514  }
515  }/*------------- end loop over potential track hits -------------------*/
516 
517  clusterVec.push_back(cluster);
518  }/*----------------- end loop over fine layers ----------------------------*/
519 
520  seedIter = hcalVec->begin();
521  }/*--------------------- end loop over AHCAL seed hits ------------------------*/
522 
523  return clusterVec;
524  }

533  EVENT::ClusterVec MipTrackFinder::selectTrackClusters(EVENT::ClusterVec clusterVec,
534  const std::string trackSelectionMode)
535  {
536  if (trackSelectionMode == "PERPENDICULAR")
537  {
538  if (_firstAhcalModulesVec.size() > 0 && _lastAhcalModulesVec.size() > 0)
539  {
540  EVENT::ClusterVec goodClusters;
541 
542  for (unsigned int iCluster = 0; iCluster < clusterVec.size(); ++iCluster)
543  {
544  int ngoodin = 0;
545  int ngoodout = 0;
546 
547  CalorimeterHitVec hitVec = clusterVec[iCluster]->getCalorimeterHits();
548  for (unsigned int iHits = 0; iHits < hitVec.size(); ++iHits)
549  {
550  const int cellID = hitVec[iHits]->getCellID0();
551  const int module = _mapper->getModuleFromCellID(cellID);
552 
553  LCIntVec::iterator searchIfFirstModule = find(_firstAhcalModulesVec.begin(),
554  _firstAhcalModulesVec.end(),
555  module);
556  LCIntVec::iterator searchIfLastModule = find(_lastAhcalModulesVec.begin(),
557  _lastAhcalModulesVec.end(),
558  module);
559 
560  if (searchIfFirstModule != _firstAhcalModulesVec.end()) ngoodin++;
561  if (searchIfLastModule != _lastAhcalModulesVec.end()) ngoodout++;
562  }/*-------- end loop over track hits ---------------------*/
563 
564  if ((ngoodin > 1) && (ngoodout > 1))
565  goodClusters.push_back(clusterVec[iCluster]);
566  else delete clusterVec[iCluster];
567 
568  }/*------------ end loop over track clusters -----------------*/
569 
570  return goodClusters;
571  }
572  else
573  {
574  streamlog_out(ERROR)<<"PERPENDICULAR tracks chosen, but no first/last AHCAL modules list given"<<endl;
575  exit(1);
576  }
577  }/*-------------------------- end of PERPENDICUALR mode --------------------------------------------*/
578 
579  else if (trackSelectionMode == "NHITS")
580  {
581  EVENT::ClusterVec goodClusters;
582  for (unsigned int iCluster = 0; iCluster < clusterVec.size(); ++iCluster)
583  {
584  CalorimeterHitVec hitVec = clusterVec[iCluster]->getCalorimeterHits();
585  if (hitVec.size() > (unsigned int)_trackNoHitsThreshold)
586  {
587  streamlog_out(DEBUG4)<<"The found track reaches the threshold of "<< _trackNoHitsThreshold
588  <<" hits and has "<<hitVec.size()<<endl;
589  goodClusters.push_back(clusterVec[iCluster]);
590  }
591  else
592  {
593  delete clusterVec[iCluster];
594  streamlog_out(DEBUG4)<<"The found track does NOT reach the threshold of "<< _trackNoHitsThreshold
595  <<" hits and has "<<hitVec.size()<<" hits"<<endl;
596  }
597  }/*----------------------- end loop over track clusters ----------------------------------------*/
598 
599  return goodClusters;
600  }/*--------------------------- end of NHITS mode ---------------------------------------------------*/
601  /*---------------------------- if no TrackSelectionMode ----------------------------------------------*/
602  else
603  {
604  streamlog_out(ERROR)<<"No TrackSelectionMode chosen, aborting..."<<endl;
605  exit(1);
606  }
607  }

614  ClusterImpl* MipTrackFinder::getWholeTrack(Cluster *trackCluster)
615  {
616  float meanXOfTrack = 0;
617  float meanYOfTrack = 0;
618 
619  CalorimeterHitVec hitVec = trackCluster->getCalorimeterHits();
620  for (unsigned int iHits = 0; iHits < hitVec.size(); ++iHits)
621  {
622  meanXOfTrack += hitVec[iHits]->getPosition()[0];
623  meanYOfTrack += hitVec[iHits]->getPosition()[1];
624  }/*------------ end loop over track clusters -----------------*/
625 
626  meanXOfTrack /= hitVec.size();
627  meanYOfTrack /= hitVec.size();
628 
629  delete trackCluster;
630 
631  streamlog_out(DEBUG4)<<" Mean X: "<<meanXOfTrack<<" mean Y: "<<meanYOfTrack<<endl;
632 
633  ClusterImpl *wholeTrack = new ClusterImpl();
634 
635  EVENT::CalorimeterHitVec::iterator iter;
636  for (iter = _allAhcalHitsInEventVec.begin(); iter != _allAhcalHitsInEventVec.end(); ++iter)
637  {
638  float differenceInX = fabs((*iter)->getPosition()[0] - meanXOfTrack);
639  float differenceInY = fabs((*iter)->getPosition()[1] - meanYOfTrack);
640 
641  streamlog_out(DEBUG4)<<" differenceInX: "<<differenceInX<<" differenceInY: "<<differenceInY<<endl;
642 
643  /*be aware of the misalignment information in the future*/
644  if (differenceInX < 16 && differenceInY < 16)
645  {
646  streamlog_out(DEBUG0)<<" Added hit to the track"<<endl;
647  wholeTrack->addHit((*iter), 1.);
648  }
649  }/*----------------- end loop over all AHCAL hits in the event*/
650 
651  return wholeTrack;
652  }
