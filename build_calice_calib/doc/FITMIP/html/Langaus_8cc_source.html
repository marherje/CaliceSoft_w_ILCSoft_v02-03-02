234 double Langaus::findePeak(){
235 
236 
237  double par[3],epar[3];
238  par[0] = _ml->getVal();
239  par[1] = _sl->getVal();
240  par[2] = _sg->getVal();
241  epar[0] = _ml->getError();
242  epar[1] = _sl->getError();
243  epar[2] = _sg->getError();
244 
245  if(_peak==0) _peak=findPeak();
246  double derivates[3];
247  for(int i=0;i<3;++i) derivates[i]=0;
248  double step;
249 
250  int count;
251  for(int i=0;i<3;i++){
252  step=0.01*par[i];
253  if(epar[i]/par[i]>0.01) step=epar[i]/4.0;
254  count=0;
255  do{
256  ++count;
257 
258  par[i]+=step;
259  derivates[i]=(findPeak(par)-_peak)/step;
260  par[i]-=step;
261 
262  if(par[i]-step>0&&i!=2){
263  par[i]-=step;
264  derivates[i]=(derivates[i]+(_peak-findPeak(par))/step)/2;
265  par[i]+=step;
266  }
267  else{
268  //derivates[i]=0;
269  }
270  if(i==2){
271  par[i]-=step;
272  derivates[i]=(derivates[i]+(_peak-findPeak(par))/step)/2;
273  par[i]+=step;
274  }
275  if(derivates[i]==0) ++step;
276  }while(derivates[i]==0&&count<20);
277  }
278  findPeak(par);
279 
280  double error=0;
281  double cor[3][3];
282  cor[1][2] = _r->correlation("sl","sg");
283  cor[0][1] = _r->correlation("ml","sl");
284  cor[0][2] = _r->correlation("ml","sg");
285  cor[2][1] = cor[1][2],cor[1][0] = cor[0][1],cor[2][0] = cor[0][2];
286  for( int i=0;i<3;i++) cor[i][i]=1;
287 
288  for(int i=0;i<3;i++){
289  for(int j=0;j<3;j++){
290  error+=derivates[i]*cor[i][j]*derivates[j]*epar[i]*epar[j];
291  }
292  }
293  if(error<0){
294  throw "Not positive definite correlation matrix";
295  }
296  error=sqrt(error);
297 
298  printf("peak:\t %.2f +- %.2f\n",_peak,error);
299 
300  _ePeak=error;
301  return error;
302 }
