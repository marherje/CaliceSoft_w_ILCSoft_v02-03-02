131  {
132  //we calculate the mean, excluding bad sensors
133  if(!isBadValue(sensorTemp[module][s])) {
134  _moduleMeanTemp[module] += sensorTemp[module][s];
135  _moduleMeanTempRMS[module] += sensorTemp[module][s] * sensorTemp[module][s];
136  _moduleMeanCount[module]++;
137

139  _caloMeanTempRMS += sensorTemp[module][s] * sensorTemp[module][s];
140  _caloMeanCount++;
141  }
142  }
143

145  _moduleMeanTemp[module] /= _moduleMeanCount[module];
146  _moduleMeanTempRMS[module] /= _moduleMeanCount[module];
147  _moduleMeanTempRMS[module] -= _moduleMeanTemp[module] * _moduleMeanTemp[module];
148  _moduleMeanTempRMS[module] = sqrt( _moduleMeanTempRMS[module] );
149 
150  //now take out the bad guys and replace'em with the mean

152  if(isBadValue(sensorTemp[module][s])) {
153  sensorTemp[module][s] = _moduleMeanTemp[module];
154  sensorTempError[module][s] = _moduleMeanTempRMS[module];
155  }

159  }
160  }

164  _caloMeanTempRMS /= _caloMeanCount;
165  }

171  sensorTempError[bad_mods[i]].end(),

178  }
179

182  // 2) set average per module to all sensors
183

201 
202  for (unsigned iLayer=2; iLayer<=37; iLayer++){
203  float temperatureForMedian[3];
204  temperatureForMedian[0] = _moduleMeanTemp[layer2module[iLayer-1]-1];
205  temperatureForMedian[1] = _moduleMeanTemp[layer2module[iLayer]-1];
206  temperatureForMedian[2] = _moduleMeanTemp[layer2module[iLayer+1]-1];

208  moduleMeanTempMedianFiltered[layer2module[iLayer]] = median;
209  }
210  moduleMeanTempMedianFiltered[layer2module[1]] = _moduleMeanTemp[layer2module[1]-1];
211  moduleMeanTempMedianFiltered[layer2module[38]] = _moduleMeanTemp[layer2module[38]-1];
212 
213  float newCaloMeanTemp = 0.;
214 
215  for( unsigned module=1; module<=38; module++ ) {
216  for( unsigned s=0; s<N_SENSORS; s++ ) {
217  sensorTemp[module-1][s] = moduleMeanTempMedianFiltered[module];
218  sensorTempError[module-1][s] = _moduleMeanTempRMS[module-1];
219  }
220  _moduleMeanTemp[module-1] = moduleMeanTempMedianFiltered[module];
221  newCaloMeanTemp += moduleMeanTempMedianFiltered[module];
222  }
