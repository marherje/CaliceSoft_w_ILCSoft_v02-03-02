171  void Alignment::calculateEffectiveTransformation() {
172  // calculate an effective translation and rotation angle
173  // such that : x_prime = Rot(alpha) x + x_trans.
174  // Note, the rotation axis is the y-axis, thus the y-component is only translated.
175 
176  // If the reference and the transformation are bitwise identical then do not rotate:
177  if (_detectorAngleZX[kDetector]==_detectorAngleZX[kReference]) {
178 
179  _detectorAngleZX[kEffective]=0.;
180  _detectorAngleZXSin=0.;
181  _detectorAngleZXCos=1.;
182 
183  }
184  else {
185  _detectorAngleZX[kEffective]=_detectorAngleZX[kDetector]-_detectorAngleZX[kReference];
186  _detectorAngleZXSin=sin(_detectorAngleZX[kEffective]);
187  _detectorAngleZXCos=cos(_detectorAngleZX[kEffective]);
188  }
189 
190  // inverse rotation :
191  double c = cos(_detectorAngleZX[kReference]);
192  double s = sin(_detectorAngleZX[kReference]);
193 
194  _detectorRotationX0[kEffective]=0;
195  _detectorRotationZ0[kEffective]=0;
196 
197  double rel_pos_x = _detectorPos[kDetector][0]+_detectorRotationX0[kDetector]
198  - _detectorPos[kReference][0]-_detectorRotationX0[kReference];
199 
200  double rel_pos_z = _detectorPos[kDetector][2]+_detectorRotationZ0[kDetector]
201  - _detectorPos[kReference][2]-_detectorRotationZ0[kReference];
202 
203  _detectorPos[kEffective][0] =
204  _detectorRotationX0[kReference]
205  + c * rel_pos_x + s * rel_pos_z
206  - _detectorAngleZXCos * _detectorRotationX0[kDetector]
207  - _detectorAngleZXSin * _detectorRotationZ0[kDetector];
208 
209  _detectorPos[kEffective][1] = _detectorPos[kDetector][1] - _detectorPos[kReference][1];
210 
211  _detectorPos[kEffective][2] =
212  _detectorRotationZ0[kReference]
213  - s * rel_pos_x + c * rel_pos_z
214  + _detectorAngleZXSin * _detectorRotationX0[kDetector]
215  - _detectorAngleZXCos * _detectorRotationZ0[kDetector];
216  }
