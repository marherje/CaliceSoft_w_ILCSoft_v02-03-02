985  DecoderSet *decoder = getDecoder();

987  unsigned int k = decoder->getKFromCellID(cellID);
988  if ( ! valid(k,_kAvailable) ) return getMaxIndex()+1;
989 
990  /* check if channel (i/j) lays inside this module
991  * NOTE: to speed up, we assume at this place, that everything was filled right
992  * we assume that when K is available, also a module

994  */
995  unsigned int ijIndex = getIndexIJ(decoder->getIFromCellID(cellID), decoder->getJFromCellID(cellID));
996  unsigned int module = _moduleVk[k];
997  unsigned int moduleType = _typeVmodule[module];
998  if ( ! valid( ijIndex, _ijAvailable[moduleType]) ) return getMaxIndex()+1;
999 
1000  // everything seems to be alright; let's return the compact index
1001  unsigned int chanIndex = _chanVij[moduleType][ijIndex];
1002  return getCompactIndex(module,getChipFromChanIndex(chanIndex),getChanFromChanIndex(chanIndex));
