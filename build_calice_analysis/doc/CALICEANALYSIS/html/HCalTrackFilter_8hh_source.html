146  EHoughBandInRect isInQuadrant(double mMin, double tMin, double mMax, double tMax)
147  {
148 
149  using namespace std;
150 
151  // check if upperLine crosses this rect
152  double upDeltaTopLeft = _upperLine.eval(mMin) - tMin;
153  double upDeltaTopRight = _upperLine.eval(mMax) - tMin;
154  double upDeltaBottomLeft = _upperLine.eval(mMin) - tMax;
155  double upDeltaBottomRight = _upperLine.eval(mMax) - tMax;
156 
157  streamlog_out(DEBUG0) << "Quadrant check "
158  << setw(8) << mMin << "-m->" << setw(8) << mMax << " "
159  << setw(8) << tMin << "-t->" << setw(8) << tMax << endl;
160 
161  streamlog_out(DEBUG0) << " Up Delta:"
162  << setw(12) << upDeltaTopLeft << " / "
163  << setw(12) << upDeltaTopRight << " / "
164  << setw(12) << upDeltaBottomLeft << " / "
165  << setw(12) << upDeltaBottomRight << " "
166  << endl;
167 
168  if (! ( (upDeltaTopLeft > 0 && upDeltaTopRight > 0 && upDeltaBottomLeft > 0 && upDeltaBottomRight > 0) ||
169  (upDeltaTopLeft < 0 && upDeltaTopRight < 0 && upDeltaBottomLeft < 0 && upDeltaBottomRight < 0) ) )
170  {
171  return UpperLineCrossing;
172  }
173 
174  // upper line did not cross this rect --> check lower line
175  double lowDeltaTopLeft = _lowerLine.eval(mMin) - tMin;
176  double lowDeltaTopRight = _lowerLine.eval(mMax) - tMin;
177  double lowDeltaBottomLeft = _lowerLine.eval(mMin) - tMax;
178  double lowDeltaBottomRight = _lowerLine.eval(mMax) - tMax;
179 
180  streamlog_out(DEBUG0) << " Low Delta:"
181  << setw(12) << lowDeltaTopLeft << " / "
182  << setw(12) << lowDeltaTopRight << " / "
183  << setw(12) << lowDeltaBottomLeft << " / "
184  << setw(12) << lowDeltaBottomRight << " "
185  << endl;
186 
187  if (! ( (lowDeltaTopLeft > 0 && lowDeltaTopRight > 0 && lowDeltaBottomLeft > 0 && lowDeltaBottomRight > 0) ||
188  (lowDeltaTopLeft < 0 && lowDeltaTopRight < 0 && lowDeltaBottomLeft < 0 && lowDeltaBottomRight < 0) ) )
189  {
190  return LowerLineCrossing;
191  }
192 
193  // no crossing between both lines, but maybe this rect is in between both
194 
195  if ( upDeltaTopLeft > 0 && upDeltaTopRight > 0 && lowDeltaBottomLeft < 0 && lowDeltaBottomRight < 0 )
196  return RectIncludedInBand;
197  else
198  return NotInRect;
199  }
