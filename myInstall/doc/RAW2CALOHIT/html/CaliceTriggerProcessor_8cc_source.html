163  void CaliceTriggerProcessor::processEvent(LCEvent* evt){
164  try {
165  //UInt_t ev_nr=evt->getEventNumber();
166  // copy fifo from the event to the trigger handler
167  getTriggerHandler()->setTrigger(evt);
168  // search around the nominal main word were the enabled trigger fired.
169  UInt_t main_word=getTriggerHandler()->searchTriggerMainWord();
170  evt->parameters().setValue(_parNameTriggerMainWord, static_cast<int>(main_word));
171 
172  if (main_word >= getTriggerHandler()->getUsedFifoDepth() ) {
173  _noMainWord++;
174  }
175 
176  // scan the history before and after the main word
177  getTriggerHandler()->searchTriggerHistory();
178 
179  CALICE::TriggerBits out_of_range_triggers=getTriggerHandler()->getOutOfRangeTriggerBits();
180  if ( out_of_range_triggers.getTriggerBits() & getTriggerHandler()->getPrincipleTriggerMask() ) {
181  _eventsWithOutOfRangeTriggers++;
182  }
183 
184 
185  //get the history prior to the main word and copy it to the event header.
186  //This history takes only bits into account which were not set in
187  //the the main word or after those bits were at least one time
188  //off before the main word
189  const std::map<unsigned int, unsigned int > &pre_trigger_history = getTriggerHandler()->getPreTriggerHistory();
190  std::vector<int> temp_pos;
191  std::vector<int> temp_bits;
192  bool dirty_event=false;
193  for (std::map<unsigned int, unsigned int>::const_iterator history_iter = pre_trigger_history.begin();
194  history_iter != pre_trigger_history.end();
195  history_iter++ ) {
196  temp_pos.push_back(history_iter->first);
197  temp_bits.push_back(history_iter->second);
198  if (history_iter->first < static_cast<unsigned int>(_noTriggerActivityRange[0]) ) {
199  if (history_iter->second & _beamTriggerBitMask) dirty_event=true;
200  }
201  }
202 
203  //Do not give the vectors to lcio if they are empty
204  //otherwise the lcio file gets corrupted !!!!???
205  if(temp_pos.size()>0) evt->parameters().setValues(_parNameTriggerPreHistoryPos , temp_pos);
206  if(temp_bits.size()>0) evt->parameters().setValues(_parNameTriggerPreHistoryBits , temp_bits);
207 
208 
209  // get the history posterior to the main word and copy it to the event header.
210  //This history takes only bits into account which were not set in
211  //the the main word or after those bits were at least one time
212  //off after the main word
213  const std::map<unsigned int, unsigned int > &post_trigger_history = getTriggerHandler()->getPostTriggerHistory();
214  temp_pos.clear();
215  temp_bits.clear();
216  for (std::map<unsigned int, unsigned int>::const_iterator history_iter = post_trigger_history.begin();
217  history_iter != post_trigger_history.end();
218  history_iter++ ) {
219  temp_pos.push_back(history_iter->first);
220  temp_bits.push_back(history_iter->second);
221 
222  if (history_iter->first < static_cast<unsigned int>(_noTriggerActivityRange[1]) ) {
223  if (history_iter->second & _beamTriggerBitMask) dirty_event=true;
224  }
225  }
226  if(temp_pos.size()>0) evt->parameters().setValues(_parNameTriggerPostHistoryPos , temp_pos);
227  if(temp_bits.size()>0) evt->parameters().setValues(_parNameTriggerPostHistoryBits , temp_bits);
228 
229 
230  CALICE::ErrorBits error(evt->getParameters().getIntVal(PAR_ERROR_BITS));
231  if (dirty_event) {
232  error.setDirtyEvent();
233  evt->parameters().setValue(PAR_ERROR_BITS,error.getBits());
234  }
235 
236 
237  /*
238 #ifdef TRIGGER_HANDLER_IS_SINGLETON
239  CALICE::TriggerBits trigger_conf(((LCEvent *) evt)->getParameters().getIntVal(_parNameTriggerConf));
240  CALICE::TriggerBits trigger_event(((LCEvent *) evt)->getParameters().getIntVal(_parNameTriggerEvent));
241 #else
242  CALICE::TriggerBits trigger_conf(getTriggerHandler()->getTriggerConfiguration());
243  CALICE::TriggerBits trigger_event(getTriggerHandler()->getTriggerEvent());
244 
245  evt->parameters().setValue(_parNameTriggerConf , trigger_conf.getTriggerBits());
246  evt->parameters().setValue(_parNameTriggerEvent, trigger_event.getTriggerBits());
247 #endif
248  */
249 
250  CALICE::TriggerBits trigger_conf(getTriggerHandler()->getTriggerConfiguration());
251  CALICE::TriggerBits trigger_andenable(getTriggerHandler()->getAndEnableBits());
252  CALICE::TriggerBits trigger_event(getTriggerHandler()->getTriggerEvent());
253 
254 
255  evt->parameters().setValue(_parNameTriggerConf , trigger_conf.getTriggerBits());
256  evt->parameters().setValue(_parNameTriggerAndEnable , trigger_andenable.getTriggerBits());
257  evt->parameters().setValue(_parNameTriggerEvent, trigger_event.getTriggerBits());
258 
259 
260 #ifdef WITH_CONTROL_HISTOGRAMS
261  histmgr::HistMgr *histogramList=histmgr::HistMgr::getInstance();
262  histmgr::HistogramCollection_t & trigger_bits_hist_arr=histogramList->getHistogramCollection(_histGroupKey,_histTriggerConfKey);
263  histmgr::HistogramCollection_t & trigger_post_hist_arr=histogramList->getHistogramCollection(_histGroupKey,_histTriggerPostKey);
264  histmgr::HistogramCollection_t & trigger_pre_hist_arr=histogramList->getHistogramCollection(_histGroupKey,_histTriggerPreKey);
265 
266  unsigned int input_enable=getTriggerHandler()->getTrgConfigurationMask();
267 
268  UInt_t main_word_content=0;
269  if (getTriggerHandler()->isTriggerFifoDepthGood()) {
270  main_word_content=getTriggerHandler()->getSmearedTriggerMainWord();
271  }
272  else {
273  // trigger fifo error
274  }
275 
276  unsigned int mask=1;
277  for (unsigned int bit_i=0; bit_i<32; bit_i++, mask<<=1) {
278  if (input_enable & mask ) trigger_bits_hist_arr.histogram(0)->fill(bit_i);
279  if (trigger_conf.getTriggerBits() & mask ) trigger_bits_hist_arr.histogram(2)->fill(bit_i);
280  if (main_word_content & mask ) trigger_bits_hist_arr.histogram(1)->fill(bit_i);
281  if (trigger_event.getTriggerBits() & mask ) trigger_bits_hist_arr.histogram(3)->fill(bit_i);
282 
283  for (std::map<unsigned int, unsigned int>::const_iterator history_iter = pre_trigger_history.begin();
284  history_iter != pre_trigger_history.end();
285  history_iter++ ) {
286  if (history_iter->second & mask ) {
287  trigger_pre_hist_arr.histogram(bit_i)->fill(history_iter->first);
288  }
289  }
290 
291  for (std::map<unsigned int, unsigned int>::const_iterator history_iter = post_trigger_history.begin();
292  history_iter != post_trigger_history.end();
293  history_iter++ ) {
294  if (history_iter->second & mask ) {
295  trigger_post_hist_arr.histogram(bit_i)->fill(history_iter->first);
296  }
297  }
298 
299 
300  }
301  if (getTriggerHandler()->isSWTriggerOn() ) {
302  trigger_bits_hist_arr.histogram(0)->fill(33);
303  trigger_bits_hist_arr.histogram(1)->fill(33);
304  }
305  if (getTriggerHandler()->isCalibrationOn() ) {
306  trigger_bits_hist_arr.histogram(0)->fill(34);
307  trigger_bits_hist_arr.histogram(1)->fill(34);
308  }
309 #endif
310  }
311  catch (std::exception &err) {
312  std::cerr << err.what() << std::endl;
313  throw err;
314  }
