233  unsigned x1_i=xBinning_.find(xMap.d2());
234  if (x0_i <= 0) x0_i = 1;
235  if (x1_i > xBinning_.n_bins()) x1_i = xBinning_.n_bins();
236 
237  unsigned y0_i=yBinning_.find(yMap.d1());
238  unsigned y1_i=yBinning_.find(yMap.d2());
239 
240  if (y0_i <= 0) y0_i=1;
241  if (y1_i > yBinning_.n_bins() ) y1_i = yBinning_.n_bins();
242 
243  unsigned int bins_per_y_slice = xBinning_.n_bins()+2;
244 
245  unsigned int bin_x0_i = y0_i*bins_per_y_slice + x0_i;
246 
247  double width_x=xBinning_.bin_width();
248  double width_y=yBinning_.bin_width();
249 
250  std::vector<Gdk::Point> polyline;
251 
252  double y = yBinning_.x_min() + (y0_i-1) * width_y;
253 
254  if ( (xMap.i2()-xMap.i1()) <= 0 || ( yMap.i1()-yMap.i2() ) <= 0) return;
255 
256  unsigned int combine_bins_x=1;
257  if ( static_cast<unsigned int>(xMap.i2()-xMap.i1()) < 10 * (x1_i - x0_i) ) {

259  if ( static_cast<unsigned int>(xMap.i2()-xMap.i1()) < 10 * (x1_i - x0_i) / combine_bins_x) {
260  combine_bins_x++;
261  }
262  }
263

313  unsigned int last_bin_y=( y_i+combine_bins_y > yBinning_.n_bins()+1 ? yBinning_.n_bins()+1 : y_i+combine_bins_y) ;
