385  void HodoscopeWriteEngine::reconstruct(int ihod) {
386  double xsum,ysum;
387 
388  for (int axis=0;axis<2;axis++) {
389  xsum=0;
390  ysum=0;
391  for(int fiber=0;fiber<16;fiber++) {
392  int ch[2];
393  chAt(fiber,ihod,axis,ch);
394  double nphsum = _hitsFill.nph[ihod][ch[0]]+_hitsFill.nph[ihod][ch[1]];
395  xsum += nphsum*Cos(thetaWith(fiber));
396  ysum += nphsum*Sin(thetaWith(fiber));
397  }
398  double theta = argWith(xsum,ysum);
399  double reco=fiberWith(theta);
400  if (theta<0) {
401  nReco[axis]=0;
402  } else {
403  nReco[axis]=(83.5-reco)/16+1;
404  }
405  for(int n=0;n<nReco[axis];n++) {
406  recoPos[axis][n]=(edgeCorrection(reco+n*16) - 41.5)*pitch;
407  //recoPos[axis][n]=(reco+n*16 - 41.5)*pitch;
408  }

411  double HodoscopeWriteEngine::trackDistance(double x1, double y1, double x2, double y2, float * pos) {
412  double z1, z2;
413  z1 = height1 * 43.3;
414  z2 = height2 * 43.3;
415 
416  double intercept[2];
417  intercept[0] = ((pos[2]-z1)*x2 + (z2-pos[2])*x1) / (z2-z1);
418  intercept[1] = ((pos[2]-z1)*y2 + (z2-pos[2])*y1) / (z2-z1);
419 
420  double distance = Sqrt((pos[0]-intercept[0])*(pos[0]-intercept[0])+(pos[1]-intercept[1])*(pos[1]-intercept[1]));
421  return distance;
422  }
