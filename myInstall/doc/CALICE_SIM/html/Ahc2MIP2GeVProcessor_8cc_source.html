121  {
122  while( getline(fIn, line) )

135  _mapK[SimDet].insert(make_pair(SimK,K));
136  }
137  else
138  {
139  // cout << "Detector number does not exist yet, create new" << endl;
140  std::map<int,int> _pairK;
141  _pairK.insert(make_pair(SimK,K));
142  _mapK.insert(make_pair(SimDet,_pairK));
143  }
144  }
145 
146  streamlog_out(MESSAGE) << "Number of Layers with changing K: " << _mapK.size() << std::endl;
147  }
