44  void TcmtGangingModifier::processHits(TempCalHitMap& hitmap) {//process hits
45 
46  //assign roles to the parameters
47  //double TimeMean = _par[62];
48  //double TimeSigma = _par[63];
49 
50  // duplicate input collection
51  TempCalHitMap input( hitmap );
52  hitmap.clear();
53 
54  // loop over simHits, creating new ganged hits
55  for(TempCalHitMap::iterator it=input.begin(); it!=input.end(); ++it ) {
56  TempCalHit& ihit = dynamic_cast<TempCalHit&>(it->second);
57  unsigned int simid = ihit.getCellID0();
58  unsigned int ilayer = simid & 0xff; // BL: K and not K-1
59  unsigned int iy = (simid >> 8) & 0xff;
60  unsigned int ix = (simid >> 16) & 0xff;
61  double energy = ihit.getTotalEnergy();
62  double time = ihit.getPrimaryTime();
63 
64  // even (odd) layers contain horizontal (vertical) strips
65  unsigned strip = ix;
66  if(ilayer%2==0) strip = iy;
67 // unsigned newid = (strip << 12) | ilayer;
68 
69  // GL: conform with Mokka CellIDEncoding: M:3,S-1:3,I:9,J:9,K-1:6
70  /* BL: no not true. Mokka CellIDEncoding is K:8,J:8,I:8 as used above,
71  * this is standard Marlin CellID encoding for CalorimeterHits.
72  * Need to subtract 1 from K or add proper encoding string to the event.
73  * Unfortunately, the design of digisim does not respect encoding
74  * strings. Therefore, I added substraction of 1 here, to be
75  * conform with the standard encoding.
76  * to-do: make digisim aware of encoding strings, to prevent
77  * problems when standards are changed.
78  */
79  unsigned newid = ((ilayer - 1) << 24);
80  if(ilayer%2!=0) newid |= (strip << 6); // vertical strip
81  if(ilayer%2==0) newid |= (strip << 15); // horizontal strip
82 
83  // create new hit
84  TempCalHitMap::iterator it2 = hitmap.find( newid );
85  if(it2==hitmap.end()) {
86  // a new hit is needed
87  TempCalHit newhit( newid, simid, energy, time );
88  hitmap.insert( pair<long long,TempCalHit>(newid,newhit) );
89  }
90  else {
91  TempCalHit& newhit = it->second;
92  newhit.addContribution( simid, energy, time );
93  }
94 
95  }
96  }//process hits
