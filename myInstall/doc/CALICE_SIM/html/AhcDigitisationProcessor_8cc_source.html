219  void AhcDigitisationProcessor::processEvent(LCEvent* evt)
220  {
221  streamlog_out(DEBUG0)<<" \n\n ===================== START event "<<evt->getEventNumber()<<endl;
222 
223  LCCollection *inputCol = NULL;
224  try{
225  inputCol = evt->getCollection( _inputColName );
226  }
227  catch ( EVENT::DataNotAvailableException &e )
228  {
229  streamlog_out(DEBUG0)<<" Collection "<<_inputColName<<" not available, skip this event"<<endl;
230  //throw marlin::SkipEventException(this);
231  return; // allow lcio event go to next processors.
232  }
233 
234  LCCollection *noiseCol = NULL;
235  try{
236  noiseCol = evt->getCollection( _noiseColName );
237  }
238  catch ( EVENT::DataNotAvailableException &e )
239  {
240  streamlog_out(DEBUG0)<<" Collection "<<_noiseColName<<" not available, skip this event"<<endl;
241  throw marlin::SkipEventException(this);
242  }
243 
244  int nHits = inputCol->getNumberOfElements();
245  streamlog_out(DEBUG0)<<" number of ganged AHC hits: "<<nHits<<endl;
246 
247  /*the Mokka encoding string*/
248  _mokkaEncodingString = inputCol->getParameters().getStringVal(LCIO::CellIDEncoding);
249 
250  /*clean up the container at the beginning of the event*/
251  delete _gangedContainer;
252 
253  /*----------------------------------------------------------------------------------
254  First step: fill the container with ganged hits, and divide the energy of the hits
255  by the MIP to GeV factor ==> E[MIPs]*/
256  this->fillGangedContainer(inputCol);
257  if (_doWriteCollectionForDebug) writeCollectionForDebug(evt, "hcalAfterGanging");
258 
259  /*---------------------------------------------------------------------------------
260  Next step: OPTICAL cross talk, still E[MIPs]
261  */
262  if(_doOpticalCrossTalk)
263  {
264  this->simulateOpticalCrossTalk();
265  if (_doWriteCollectionForDebug) writeCollectionForDebug(evt, "hcalAfterCrossTalk");
266  }
267 
268  /*---------------------------------------------------------------------------------
269  Next step: Simulate SiPM behaviour,
270  i.e. convert the energy from MIPs to ADC counts
271  */
272  this->simulateSiPMBehaviour();
273  if (_doWriteCollectionForDebug) writeCollectionForDebug(evt, "hcalAfterSiPMBehaviour");
274 
275  /*---------------------------------------------------------------------------------
276  Next step: Add the noise from data, E[ADC]
277  */
278  if(_doAddNoise) this->addNoise(noiseCol);
279 
280  /*---------------------------------------------------------------------------------
281  Next step: Finally, create the output collection, E[ADC]
282  */
283  this->createOutputCollection(evt);
284  }

290  void AhcDigitisationProcessor::fillGangedContainer(LCCollection *inputCol)
291  {
292  streamlog_out(DEBUG0)<<"\n\n Start to fill ganged container "<<std::endl;
293 
294  /*create a new map container. Note the 'false' flag, which tells the container
295  that he is not responsible for the deletion of the contained elements.
296  We use this because at the end of the event we have all the elements from
297  the container into an LCCollection, and Marlin deletes them at the end of the event.*/
298  _gangedContainer = new MappedContainer<CalorimeterHitImpl>(_mapper, false);
299  _gangedContainer->getDecoder()->setCellIDEncoding(_mokkaEncodingString);
300 
301  /*Copy the input collection into the container, which we'll modify afterwards*/
302  for (int iHit = 0; iHit < inputCol->getNumberOfElements(); ++iHit)
303  {
304  CalorimeterHit *hit = dynamic_cast<CalorimeterHit*>(inputCol->getElementAt(iHit));
305  int trueCellID = hit->getCellID0();
306 
307  CalorimeterHitImpl *newHit = new CalorimeterHitImpl();
308  /*Convert the energy from GeV to MIPs*/
309  newHit->setEnergy(hit->getEnergy()/_mipPerGeVFactor);
310  newHit->setCellID0(trueCellID);
311  _gangedContainer->fillByCellID(trueCellID, newHit);
312  }
313 
314  streamlog_out(DEBUG0)<<"\n After ganging: gangedContainer has "
315  <<_gangedContainer->getAllElements().size()
316  <<" elements"<<endl;
317 
318  }
319 
320 
321  /*********************************************************************************/
322  /* */
323  /* Note: during cross-talk, 10% of the energy leaks to the neighbouring */
324  /* tiles, but the leaked energy is NOT subtracted from the energy because */
325  /* the measured MIP is actually only 90%MIPs. For more details, see page 101 in */
326  /* http://www-library.desy.de/preparch/desy/thesis/desy-thesis-10-006.pdf */
327  /* */
328  /*********************************************************************************/
329  void AhcDigitisationProcessor::simulateOpticalCrossTalk()
330  {
331  streamlog_out(DEBUG0)<<"\n\n Start to simulate optical cross-talk";
332 
333  std::vector<CalorimeterHitImpl*> gangedVec = _gangedContainer->getAllElements();
334  streamlog_out(DEBUG0)<<" for "<<gangedVec.size()<<" ganged hits"<<std::endl;
335 
336  /*counter for debug purposes*/
337  int countNewCells = 0;
338  int countGanged = 0;
339 
340  // Generate a new vector to save the original hits information
341  // Loop over the original hits, and add optical xtalk into the
342  // neighbours cell.
343  // It is only first order optical xtalk has been applied here.
344  std::vector<CalorimeterHitImpl> originalgangedVec;
345  for (std::vector<CalorimeterHitImpl *>::iterator iter = gangedVec.begin();
346  iter != gangedVec.end(); ++iter)
347  {
348  CalorimeterHitImpl *currentHit = (*iter);
349  originalgangedVec.push_back((*currentHit));
350  }
351 
352  /*------------------------------------------------------------------
353  Start to loop over original ganged hits
354  ------------------------------------------------------------------*/
355  for (std::vector<CalorimeterHitImpl>::iterator iter = originalgangedVec.begin();
356  iter != originalgangedVec.end(); ++iter)
357  {
358  /*set the encoding string back to the Mokka encoding string*/
359  _gangedContainer->getDecoder()->setCellIDEncoding(_mokkaEncodingString);
360 
361  int cellID = (*iter).getCellID0();
362  CalorimeterHitImpl *currentHit = &(*iter);
363 
364  _cellDescriptions->getDecoder()->setCellIDEncoding(_mokkaEncodingString);
365  CellDescription* currentCellDescription = _cellDescriptions->getByCellID(cellID);
366  const float currentTileSize = currentCellDescription->getSizeX();
367 
368  streamlog_out(DEBUG0)<<"------------"<<endl;
369  streamlog_out(DEBUG0)<<"i="<<countGanged<<" ganged I/J/K: "<< _gangedContainer->getDecoder()->getIFromCellID(cellID)
370  <<"/"<<_gangedContainer->getDecoder()->getJFromCellID(cellID)
371  <<"/"<<_gangedContainer->getDecoder()->getKFromCellID(cellID)
372  <<" currentHit: "<<currentHit
373  <<" cellID: "<<cellID
374  <<" energy[MIPs]: "<<currentHit->getEnergy()
375  <<" currentTileSize: "<<currentTileSize
376  <<endl;
377  countGanged ++;
378 
379  /*start loop over neighbours*/
380  const std::vector<int> neighboursVec =
381  _cellNeighbours->getByCellID(cellID)->getNeighbours(CALICE::CellNeighbours::direct,
382  CALICE::CellNeighbours::module);
383  streamlog_out(DEBUG0)<<" ->number of neighbours found: "<<neighboursVec.size()<<endl;
384 
385  /*before getting the neighbours, set the encoding string to the string used to
386  set the neighbours*/
387  const std::string neighboursEncodingString =
388  CALICE::CellNeighboursProcessor::getEncodingString(_cellNeighboursProcessorName);
389  streamlog_out(DEBUG0)<<" neighboursEncodingString: "<<neighboursEncodingString<<endl;
390 
391  /*---------------------------------------------------------------
392  loop over neighbours
393  -----------------------------------------------------------------*/
394  for (unsigned int i = 0; i < neighboursVec.size(); ++i)
395  {
396  streamlog_out(DEBUG0)<<" \n i="<<i<<endl;
397  _cellDescriptions->getDecoder()->setCellIDEncoding(neighboursEncodingString);
398  streamlog_out(DEBUG0)<<" neighboursEncodingString: "<<neighboursEncodingString<<endl;
399 
400  streamlog_out(DEBUG0)<<" neighbours: I/J/K="
401  <<_cellDescriptions->getDecoder()->getIFromCellID(neighboursVec[i])
402  <<"/"<<_cellDescriptions->getDecoder()->getJFromCellID(neighboursVec[i])
403  <<"/"<<_cellDescriptions->getDecoder()->getKFromCellID(neighboursVec[i])
404  <<" endcoding string: "<<_cellDescriptions->getDecoder()->getCellIDEncoding()
405  <<endl;
406 
407  CellDescription *neighbourCellDescription = _cellDescriptions->getByCellID(neighboursVec[i]);
408  if (neighbourCellDescription == NULL)
409  {
410  streamlog_out(DEBUG0)<<" sorry, no cell description found for neighbour "<<i<<endl;
411  continue;
412  }
413 
414  const float neighbourTileSize = neighbourCellDescription->getSizeX();
415  streamlog_out(DEBUG0)<<" neighbourTileSize: "<<neighbourTileSize<<endl;
416 
417  /*Define fraction of leakage going into neighbouring cells;
418  for example, if the current cell has neighbours of the same size,
419  and if the total leakage is 10%, each neighbour will get
420  an energy fraction of (leakageFraction * totalLeakage) = 0.25 * 0.10 = 0.025*/
421  float leakageFraction = 1.0/4.0;
422 
423  /*if the neighbours are larger or equal to the current tile,
424  they will get a quarter (1/4) of the leaked energy of the current tile;
425  else, the leakageFraction has to consider the smaller size of the neighbours,
426  since for the leakage is equally distributed at each side of the current tile*/
427  if (neighbourTileSize >= currentTileSize) leakageFraction *= 1;
428  else if (neighbourTileSize < currentTileSize) leakageFraction *= neighbourTileSize/currentTileSize;
429 
430  /*now we can calculate the energy which leaks from the current tile to the neighbours;
431  the factor A=(1. - _lightLeakage)
432  plays the role of an effective MIP/GeV factor. We do this because, for the muon case,
433  if we sum up the energy, we need to get the 1 MIP.
434  */
435  float leakedEnergy = leakageFraction * _lightLeakage * (currentHit->getEnergy())
436  /(1. - _lightLeakage);
437 
438  streamlog_out(DEBUG0)<<" lightLeakage: "<<_lightLeakage<<" energy: "<<currentHit->getEnergy()<<std::endl;
439  streamlog_out(DEBUG0)<<" leakageFraction: "<<leakageFraction<<" leakedEnergy: "<<leakedEnergy<<endl;
440  streamlog_out(DEBUG0)<<" looking in list for I/J/K="
441  <<_cellDescriptions->getDecoder()->getIFromCellID(neighboursVec[i])
442  <<"/"<<_cellDescriptions->getDecoder()->getJFromCellID(neighboursVec[i])
443  <<"/"<<_cellDescriptions->getDecoder()->getKFromCellID(neighboursVec[i])
444  <<endl;
445 
446  _gangedContainer->getDecoder()->setCellIDEncoding(neighboursEncodingString);
447  CalorimeterHitImpl *neighbourHit = _gangedContainer->getByCellID(neighboursVec[i]);
448 
449  if (neighbourHit == NULL)
450  {
451  neighbourHit = new CalorimeterHitImpl();
452  neighbourHit->setEnergy(leakedEnergy);
453 
454  countNewCells++;
455  streamlog_out(DEBUG0)<<" new hit, leaked energy in the neighbour: "<<neighbourHit->getEnergy()<<std::endl;
456 
457  /*need to get the cell ID in the Mokka encoding, to be able to fill the gangedContainer*/
458  const int I = _gangedContainer->getDecoder()->getIFromCellID(neighboursVec[i]);
459  const int J = _gangedContainer->getDecoder()->getJFromCellID(neighboursVec[i]);
460  const int K = _gangedContainer->getDecoder()->getKFromCellID(neighboursVec[i]);
461 
462  _gangedContainer->getDecoder()->setCellIDEncoding(_mokkaEncodingString);
463  const int cellIDinMokkaEncoding = _gangedContainer->getDecoder()->getCellID(I, J, K);
464  neighbourHit->setCellID0(cellIDinMokkaEncoding);
465  _gangedContainer->fillByCellID(cellIDinMokkaEncoding, neighbourHit);
466  }
467  else
468  {
469  streamlog_out(DEBUG0)<<" Guess what, hit alread exists, energy="<<neighbourHit->getEnergy()<<endl;
470 
471  /*add up the leakedEnergy to the original energy of the neighbours*/
472  neighbourHit->setEnergy(neighbourHit->getEnergy() + leakedEnergy);
473  }
474 
475  streamlog_out(DEBUG0)<<" final energy in the neighbour: "<<neighbourHit->getEnergy()<<std::endl;
476  }/*---------------- end loop over neighbours vector -----------------------*/
477 
478  }/*------------------- end loop over ganged iterator ----------------------------*/
479 
480 
481  /*************************************************************/
482  streamlog_out(DEBUG0)<<"\n After cross-talk: gangedContainer has "
483  <<_gangedContainer->getAllElements().size()
484  <<"elements, from which "<<countNewCells<<" new hits"
485  <<endl;
486 
487  }

494  void AhcDigitisationProcessor::simulateSiPMBehaviour()
495  {
496  streamlog_out(DEBUG0)<<"\n\n Start to simulate SiPM behaviour "<<endl;
497 
498  const std::string neighboursEncodingString =
499  CALICE::CellNeighboursProcessor::getEncodingString(_cellNeighboursProcessorName);
500 
501  /*now the ganged container contains ganged cells, with cross-talk simulated*/
502  std::vector<CalorimeterHitImpl*> gangedVec = _gangedContainer->getAllElements();
503  for (std::vector<CalorimeterHitImpl*>::iterator iter = gangedVec.begin();
504  iter != gangedVec.end(); ++iter)
505  {
506  int cellID = (*iter)->getCellID0();
507  CalorimeterHitImpl *currentHit = (*iter);
508  float energyInMIPs = currentHit->getEnergy();
509 
510  _gangedContainer->getDecoder()->setCellIDEncoding(_mokkaEncodingString);
511 
512  streamlog_out(DEBUG0)<<"\n---------------"<<endl;
513  streamlog_out(DEBUG0)<<" currentHit with cell ID "<<cellID<<", energyInMIPs "<<energyInMIPs
514  <<", I/J/K "<<_gangedContainer->getDecoder()->getIFromCellID(cellID)
515  <<"/"<<_gangedContainer->getDecoder()->getJFromCellID(cellID)
516  <<"/"<<_gangedContainer->getDecoder()->getKFromCellID(cellID)
517  <<endl;
518 
519  streamlog_out(DEBUG0)<<" calibrationEncodingString: "<<_calibContainer->getDecoder()->getCellIDEncoding()<<endl;
520  SiPMCalibrations *calibration = _calibContainer->getByCellID(cellID);
521  if (calibration == NULL)
522  {
523  streamlog_out(DEBUG0)<<" Sorry, no calibration found for cellID "<<cellID<<endl;
524  continue;
525  }
526 
527  /*-------------------- calibration coefficients --------------*/
528  LinearFitCompound *lfcMIP = calibration->getMIP();
529  LinearFitCompound *lfcGain = calibration->getGain();
530  SimpleValue *interCalibContainer = calibration->getInterCalibration();
531  SatCorrFunction *saturationFunction = calibration->getSaturationCorrection();
532 
533  /*---------------------- temperature -------------------------*/
534  SimpleValue *temperatureObject = calibration->getTemperature();
535  const float temperature = temperatureObject->getValue();
536 
537  /*---------------------- MIP ---------------------------------*/
538  float mipValue = 0;
539  if (_doMipTempCorr) mipValue = lfcMIP->eval(temperature);
540  else
541  mipValue = lfcMIP->getConstant();
542 
543  if (mipValue <= 0)
544  {
545  /*
546  streamlog_out(ERROR0) << " ERROR - ignoring cell with MIP coefficient "<<mipValue
547  << "at I/J/K "<<_gangedContainer->getDecoder()->getIFromCellID(cellID)
548  <<"/"<<_gangedContainer->getDecoder()->getJFromCellID(cellID)
549  <<"/"<<_gangedContainer->getDecoder()->getKFromCellID(cellID)
550  << endl;
551  */
552  continue;
553  }
554  if (mipValue > 2000) /*is the default value 100000?*/
555  {
556  /*
557  streamlog_out(WARNING)<<" Warning, very high MIP value, equal to "<<mipValue
558  <<", setting it to zero..."<<std::endl;
559  */
560  mipValue = 0;
561  }
562 
563  /*---------------------- gain ---------------------------------*/
564  float gainValue = 0;
565  if (_doGainTempCorr) gainValue = lfcGain->eval(temperature);
566  else
567  gainValue = lfcGain->getConstant();
568  if (gainValue <= 0)
569  {
570  /*
571  streamlog_out(ERROR0) << " ERROR - ignoring cell with GAIN coefficient "<<gainValue
572  << "at I/J/K "<<_gangedContainer->getDecoder()->getIFromCellID(cellID)
573  <<"/"<<_gangedContainer->getDecoder()->getJFromCellID(cellID)
574  <<"/"<<_gangedContainer->getDecoder()->getKFromCellID(cellID)
575  << endl;
576  */
577  continue;
578  }
579 
580  /*---------------------- InterCalibration ----------------------*/
581  float interCalibrationValue = interCalibContainer->getValue();
582  if (interCalibrationValue <= 0)
583  {
584  /*
585  streamlog_out(ERROR0) << " ERROR - ignoring cell with IC coefficient "<<interCalibrationValue
586  << "at I/J/K "<<_gangedContainer->getDecoder()->getIFromCellID(cellID)
587  <<"/"<<_gangedContainer->getDecoder()->getJFromCellID(cellID)
588  <<"/"<<_gangedContainer->getDecoder()->getKFromCellID(cellID)
589  << endl;
590  */
591  continue;
592  }
593 
594  streamlog_out(DEBUG0) << " - MIP: " << mipValue << endl;
595  streamlog_out(DEBUG0) << " - gain: " << gainValue << endl;
596  streamlog_out(DEBUG0) << " - intercalibration: " << interCalibrationValue << endl;
597 
598  //check if gainConstant or IC is default => put gainConst to have fixed LY.
599  //This prevents an event to have the "strange" saturated amplitudes
600  const SiPMCalibrationStatusBits bits = SiPMCalibrationStatusBits( calibration->getStatus() );
601 
602  if ( _correctDefaultGainToLY && (bits.gainConstantIsDefault() || bits.interCalibrationIsDefault()) ) {
603  gainValue = mipValue * interCalibrationValue / _fixedLY;
604  }
605 
606  /*---------------------- light yield ---------------------------*/
607  const float lightYield = mipValue/gainValue * interCalibrationValue;
608  const float linearPixel = lightYield * energyInMIPs;
609 
610  double saturatedPixel = 0.0;
611 
612  if(_doSaturation == true) //saturatedPixel true as default
613  {
614  saturatedPixel = saturationFunction->saturate(linearPixel);
615 
616  streamlog_out(DEBUG0)<<" - lightYield: "<<lightYield<<endl;
