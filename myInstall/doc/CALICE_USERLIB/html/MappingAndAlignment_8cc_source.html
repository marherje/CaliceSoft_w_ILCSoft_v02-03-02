351  Int_t MappingAndAlignment::getModuleIndexFromCellIndex(UInt_t cellindex) {
352  //first maskout the relevant Mokka indices K (layer number) and M (for the SiW Ecal, Wafer Row)
353  UInt_t lookupindex = ( ((cellindex & MASK_M) -1) | (cellindex & MASK_K) );
354 #ifdef RECO_DEBUG
355  std::cout << "Lookup Index: " << std::hex << lookupindex << std::dec << std::endl;
356 #endif
357  MapCelltoModuleIndex_t::iterator map_iter = _mapCelltoModuleIndex.find(lookupindex);
358  //if we have found the module_index. fine - Return it
359  if ( map_iter != _mapCelltoModuleIndex.end()) return map_iter->second;
360 
361  //if not we might be still one or more rows above
362  //the one containing the index of the lower left cell, so keep on reducing the
363  //row index
364  UInt_t isearch(0);
365  UInt_t testindex(0);
366  //Loop until we find an index;
367  while ( map_iter == _mapCelltoModuleIndex.end() ) {
368  isearch++;
369  Int_t m_index = static_cast<Int_t>(( (cellindex & MASK_M) >> SHIFT_M) - isearch - 1);
370  if(m_index < 0) {
371  stringstream message;
372  message << "MappingAndAlignment::getModuleIndexFromCellIndex:Index of lower left cell not found";
373  message << "for mokka index " << cellindex << std::endl;
374  throw std::runtime_error(message.str());
375  }
376  testindex = (static_cast<UInt_t>(m_index) << SHIFT_M) | (cellindex & MASK_K) ;
377 #ifdef RECO_DEBUG
378  std::cout << "Test Index: " << std::hex << testindex << std::dec << std::endl;
379 #endif
380  map_iter = _mapCelltoModuleIndex.find(testindex);
381 
382  }
383 #ifdef RECO_DEBUG
384  std::cout << "Before second return: " << std::endl;
385 #endif
386  return map_iter->second;
387 }

457  void MappingAndAlignment::determineMaximumCellNumberPerFrontEnd()
458  {
459  if ( isModuleConditionsDataComplete() ) {
460  _maxNCells=0;
461  for (CrateList_t::const_iterator crate_iter=_crateList.begin();
462  crate_iter!=_crateList.end();
463  crate_iter++) {
464  for (UInt_t slot_i=0; slot_i<crate_iter->second.size(); slot_i++) {
465  if (crate_iter->second[slot_i].size()>0) {
466 
467  for (UInt_t fe_i=0; fe_i+1<crate_iter->second[slot_i].size(); fe_i+=2) {
468  UInt_t n_cells[2]={0,0};
469  UInt_t last_module_index=_moduleLocationList.size();
470  for (UInt_t side_i=0; side_i<2; side_i++) {
471  UInt_t eff_fe_i=fe_i+side_i;
472  if (crate_iter->second[slot_i][eff_fe_i]<_moduleLocationList.size()) {
473  UInt_t module_index=crate_iter->second[slot_i][eff_fe_i];
474  UInt_t module_type_i=_moduleLocationList[crate_iter->second[slot_i][eff_fe_i]].second.getModuleType();
475  if ( module_type_i < _moduleTypeList.size()) {
476  if (side_i==0 || module_index != last_module_index) {
477  n_cells[side_i]=_moduleTypeList[ module_type_i ].getNCells();
478  }
479  }
480  last_module_index=module_index;
481  }
482  }
483  if (n_cells[1]>0) {
484  // if there is a left side
485  // then the maximum number of cells is
486  // defined large enough such that half of the buffer which is set to the maximum cell size
487  // will be large enough to fit the cells of the right and the left side.
488  if (n_cells[0]*2>_maxNCells) {
489  _maxNCells=n_cells[0]*2;
490  }
491  if (n_cells[1]*2>_maxNCells) {
492  _maxNCells=n_cells[0]*2;
493  }
494  }
495  else {
496  // if there is only the right side
497  if (n_cells[0]>_maxNCells) {
498  _maxNCells=n_cells[0];
499  }
500  }
501  }
502  }
503  }
504  }

509  void MappingAndAlignment::correctFullHalfConnectorErrors()
510  {
511  if (_moduleTypeList.size()==0) {
512  // No modules type have been defined yet cannot guess the required connector type (full/half).
513  return;
514  }
515  // bool changed_something=false;
516  for (CrateList_t::iterator crate_iter=_crateList.begin();
517  crate_iter!=_crateList.end();
518  crate_iter++) {
519  for (UInt_t slot_i=0; slot_i<crate_iter->second.size(); slot_i++) {
520  if (crate_iter->second[slot_i].size()>0) {
521  for (UInt_t fe_i=0; fe_i<crate_iter->second[slot_i].size(); fe_i+=2) {
522  if (isModule(crate_iter->second[slot_i][fe_i])) {
523  if (_moduleLocationList[crate_iter->second[slot_i][fe_i]].second.getModuleType()>=_moduleTypeList.size()) {
524  std::cerr << "Unknown module type connected to crate " << crate_iter->first
525  << " slot=" << slot_i << " fe=" << fe_i << std::endl;
526  continue;
527  }
528  const ModuleDescription &module_type=_moduleTypeList[_moduleLocationList[crate_iter->second[slot_i][fe_i]].second.getModuleType()];
529 
530  bool module_probably_needs_full_connector=( ( module_type.getNCells() >108) ? true : false);
531  if (isModule(crate_iter->second[slot_i][fe_i+1])) {
532  if (module_probably_needs_full_connector && crate_iter->second[slot_i][fe_i+1] != crate_iter->second[slot_i][fe_i]) {
533  std::cerr << "ERROR::Module type \"" << module_type.getModuleTypeName() << "\" (" << module_type.getNCells()
