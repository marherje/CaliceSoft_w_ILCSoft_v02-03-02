105  vector<int> bad_mods;

108  CALICE::AhcSlowReadoutModBlock mb( _sroModCol->getElementAt(i) );
109 
110  int module = mb.getModuleNumber()-1;
111  _moduleMeanTemp[module]=0;
112  _moduleMeanTempRMS[module]=0;
113  _moduleMeanCount[module]=0;
114 
115  for(unsigned s=0; s<N_SENSORS && s<mb.getCmbTemperatures().size(); s++)
116  {
117  //we calculate the mean, excluding bad sensors
118  if(!isBadValue(sensorTemp[module][s])) {
119  _moduleMeanTemp[module] += sensorTemp[module][s];
120  _moduleMeanTempRMS[module] += sensorTemp[module][s] * sensorTemp[module][s];
121  _moduleMeanCount[module]++;
122 
123  _caloMeanTemp += sensorTemp[module][s];
124  _caloMeanTempRMS += sensorTemp[module][s] * sensorTemp[module][s];
125  _caloMeanCount++;
126  }
127  }
128

131  _moduleMeanTempRMS[module] /= _moduleMeanCount[module];
132  _moduleMeanTempRMS[module] -= _moduleMeanTemp[module] * _moduleMeanTemp[module];
133  _moduleMeanTempRMS[module] = sqrt( _moduleMeanTempRMS[module] );
134

139  sensorTempError[module][s] = _moduleMeanTempRMS[module];
