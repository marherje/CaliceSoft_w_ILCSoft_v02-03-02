149  BmlSlowRunDataBlock& setTimeStamp(struct tm* timestamp)
150  {
151 
152  if(timestamp){
153  //Timestamp year month day
154  int timestampymd(0);
155  timestampymd = timestampymd | ( ( (timestamp->tm_year + TM_YEARSTART) & 0xffff) << YEARSHIFT)
156  | ( ( (timestamp->tm_mon + 1) & 0xff) << MONTHSHIFT)
157  | ( ( timestamp->tm_mday & 0xff) << DAYSHIFT);
158 
159 
160  //Timestamp hour minute second
161  int timestamphms(0);
162  timestamphms = timestamphms | ( (timestamp->tm_hour & 0xff) << HOURSHIFT)
163  | ( (timestamp->tm_min & 0xff) << MINUTESHIFT)
164  | ( (timestamp->tm_sec & 0xff) << SECONDSHIFT);
165 
166 
167  obj()->setIntVal(kIntBmlSroRunPosTimeStampymd, timestampymd);
168  obj()->setIntVal(kIntBmlSroRunPosTimeStamphms, timestamphms);
169  } else {
170  obj()->setIntVal(kIntBmlSroRunPosTimeStampymd, 0);
171  obj()->setIntVal(kIntBmlSroRunPosTimeStamphms, 0);
172 
173  }
174  return *this;
175  }

180  LCTime getTimeStamp() const
181  {
182  int timestampymd(getIntVal(kIntBmlSroRunPosTimeStampymd));
183  int timestamphms(getIntVal(kIntBmlSroRunPosTimeStamphms));
184 
185  //composition of an intermediate LCTime object in order
186  //to get everything up to seconds in nanosecs.
187  LCTime init( (timestampymd >> YEARSHIFT) & 0xffff,
188  (timestampymd >> MONTHSHIFT) & 0xff,
189  (timestampymd >> DAYSHIFT) & 0xff,
190  (timestamphms >> HOURSHIFT) & 0xff,
191  (timestamphms >> MINUTESHIFT) & 0xff,
192  (timestamphms >> SECONDSHIFT) & 0xff );
193  LCTime finaltime(init.timeStamp());
194  return finaltime;
