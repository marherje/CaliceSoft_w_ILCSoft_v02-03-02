96  // enforce an allocation of the int and float array;

98  UInt_t name_start = 0;
99  switch (options) {
100  case kModuleDescriptionNormal: {
101  // hack to reserve space: set the last needed cell to a dummy value. This will automatically resize the underlying storage to
102  // the needed size.
103  _cellIndexStart=kNModuleDescriptionIntsOrNCells;
104  _obj->setIntVal(_cellIndexStart + n_cells + n_string_buffer_size -1,0);
105  _obj->setIntVal(kModuleDescriptionIntModuleType,static_cast<int>(module_type));
106  // In normal ModuleDescription objects only elements until kNModuleDescriptionIntsOrNCells are needed plus one int
107  // per n_cells + buffer for the module name
108  name_start = static_cast<int>(kNModuleDescriptionIntsOrNCells+n_cells);
109  _obj->setIntVal(kModuleDescriptionIntNameIndexOrMinusOne,name_start);
110 
111  _obj->setFloatVal(kNModuleDescriptionFloats+n_cells*2 -1 , 0. );
112  break;
113  }
114  case kModuleDescriptionWithCellDimensionAndZPos:
115  _cellZPosStart=n_cells*4+kNModuleDescriptionFloats;
116  // reserve more float space (see above):
117  _obj->setFloatVal(_cellZPosStart+n_cells -1 , 0. );
118 
119  case kModuleDescriptionWithCellDimension: {
120  _cellIndexStart=kNModuleDescriptionInts;
121  _cellDimensionStart=n_cells*2+kNModuleDescriptionFloats;
122  // reserve space (see above):
123  _obj->setIntVal( _cellIndexStart + n_cells + n_string_buffer_size -1 ,0);
124  // reserve float space (see above):
125  _obj->setFloatVal(_cellDimensionStart+n_cells*2 -1 , 0. );
126 
127  _obj->setIntVal(kModuleDescriptionIntModuleType,static_cast<int>(module_type));
128  _obj->setIntVal(kNModuleDescriptionIntsOrNCells,n_cells);
