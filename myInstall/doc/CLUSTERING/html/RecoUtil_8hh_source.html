558  this->nxt=lnk;

572  while(1){
573  while(clnk[cdepth]->nxt){
574  clnk[cdepth]=clnk[cdepth]->nxt;
575  if(_find_first(cdepth))
576  return; // success
577  }
578  // try go right
579  clg[cdepth]=clg[cdepth]->nxt;
580  if(!clg[cdepth])
581  break;
582  clnk[cdepth]=clg[cdepth]->cld;
583  if(_find_first(cdepth))
584  return;
