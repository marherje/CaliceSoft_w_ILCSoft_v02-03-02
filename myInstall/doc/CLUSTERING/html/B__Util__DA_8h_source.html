265 /* Find M smallest/biggest values (ordered) in array with size N.
266  * Works faster than qsort in case N<<M.
267  *
268  * b[M] is filled with sordered values
269  * ia[M] is filled with indexes (in a) of this values.
270  * both arrays are optional (but must provide required space when used)
271  *
272  * Inspirated by: V.L. Morgunov
273  * Implemented by: A. Zhelezov
274  */
275 template<typename _t,typename _func>
276  void SortMfromN(unsigned n,const _t *a,unsigned m,_t *b=0,unsigned *ia=0){
277  if(m>n || !a){
278  std::cerr<<"Incorrect Min_amp call: "<<n<<" "<<m<<std::endl;
279  return;
280  }
281  typedef struct _item {
282  struct _item *nxt;
283  struct _item *prv;
284  unsigned idx;
285  } item;
286  item x[m];
287  unsigned x_used=0;
288  item *head=0,*last=0,*p;
289  if(m>1){
290  for(unsigned i=0;i<n;i++){
291  p=head;
292  for(;p;p=p->nxt){
293  if(_func::comp(a[i],a[p->idx]))
294  continue;
295  item *newp;
296  if(x_used<m)
297  newp=x+x_used++;
298  else if(p == last){
299  last->idx=i;
300  break;
301  } else {
302  newp=last;
303  last=last->prv;
304  last->nxt=0;
305  }
306  newp->nxt=p;
307  if((newp->prv=p->prv))
308  p->prv->nxt=newp;
309  else
310  head=newp;
311  p->prv=newp;
312  newp->idx=i;
313  break;
314  }
315  if(!p && x_used<m){
316  item *newp=x+x_used++;
317  if((newp->prv=last))
318  last->nxt=newp;
319  else
320  head=newp;
321  last=newp;
322  newp->nxt=0;
323  newp->idx=i;
324  }
325  }
326  } else if(m==1){ // make result collection work in this case
327  head=x;
328  x[0].idx=0;
329  x[0].nxt=0;
330  }
331  if(b){
332  p=head;
333  for(unsigned i=0;i<m;i++,p=p->nxt)
334  b[i]=a[p->idx];
335  }
336  if(ia){
337  p=head;
338  for(unsigned i=0;i<m;i++,p=p->nxt)
339  ia[i]=p->idx;
340  }
341 }
