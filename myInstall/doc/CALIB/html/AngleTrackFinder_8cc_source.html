379  /* here apply criteria to use the number of hits in HCAL to apply the muon*/

384  streamlog_out(DEBUG4) << "Using ECAL and TCMT information" << endl;

386  /* ECAL information*/
387  std::pair<int, float> nhits_and_esum(extractNumberOfHitsAndEnergySum(_ecalInputCol));
388  int ecalNumberOfHits = nhits_and_esum.first;
389 
390  if ((ecalNumberOfHits > _ecalMinNoHits) && (ecalNumberOfHits < _ecalMaxNoHits))
391  {
392  analyse = true;
393  streamlog_out(DEBUG4) << "ECAL criteria reached. "
394  << "Going to do further processing." << endl;
395  }
396  else
397  {
398  analyse = false;
399  streamlog_out(DEBUG4) << "ECAL criteria *not* reached. "
400  << "Event will be skipped." << endl;
401  }
402  }
403  else if (_useTCMT == true)
404  {
405  // TCMT information
406  std::pair<int, float> nhits_and_esum = extractNumberOfHitsAndEnergySum(_tcmtInputCol);
407  float tcmtEnergySum = nhits_and_esum.second;
408 
409  if((tcmtEnergySum > _tcmtMinEnergySum))
410  {
411  analyse = true;
412  streamlog_out(DEBUG4) << "TCMT criteria reached. "
413  << "Going to do further processing." << endl;
414  }
415  else
416  {
417  analyse = false;
418  streamlog_out(DEBUG4) << "TCMT criteria *not* reached. "
419  << "Event will be skipped." << endl;
420  }
421  }
422  else
423  {
424  if (_isNonMuonRun == true)
425  {
426  /* this is just for non-muon runs case!*/
427  std::pair<int, float> nhits_and_esum(extractNumberOfHitsAndEnergySum(_ahcalInputCol));
428  int nhits_hcal = nhits_and_esum.first;
429  streamlog_out(DEBUG4) << "hcal hits " << nhits_hcal << endl;
430 
431  // FIXME this check makes no sense here since there are always hits in every cell
432  // check is done in findTrack instead
433  if (nhits_hcal < _ahcalMaxNoHits)
434  {
435  analyse = true;
436  }
437  else
438  {
439  analyse = true;
440  }
441  }
442  else
443  {
444  analyse = true;
445  }
446  streamlog_out(DEBUG4) << " ->AHCAL only analysis will start." << endl;
447  }
448 
449  return analyse;
450  }

481 
482  streamlog_out(DEBUG4) << " haveListOfBadModules: " << haveListOfBadModules << endl;
483 
484  LCTypedVector<CalorimeterHit> hcalVec(_ahcalInputCol);
485  _allAhcalHitsInEventVec = hcalVec;
486  std::vector<CalorimeterHit*> selectedHcalHitsVec;
487 
488  /*counter for debug purposes*/
489  int counter = 0;
490 
491  for (unsigned int iHits = 0; iHits < hcalVec.size(); ++iHits)
492  {
493  const int cellID = hcalVec[iHits]->getCellID0();
494  const int module = _mapper->getModuleFromCellID(cellID);
495 
496  bool hitIsInBadModule = false;
497 
498  if (haveListOfBadModules == true)
499  {
500  if (std::find(_badAhcalModulesVec.begin(), _badAhcalModulesVec.end(), module) != _badAhcalModulesVec.end())
501  hitIsInBadModule = true;
502 
503  if (hitIsInBadModule)
504  {
505  streamlog_out(DEBUG4) << " -> " << counter << " hit in module "
506  << module << "-> on list of bad modules" << endl;
507  }
508  else
509  {
510  streamlog_out(DEBUG4) << " -> " << counter << " hit in module "
511  << module << "-> NOT on list of bad modules" << endl;
512  }
513  }
514 
515  // get the amplitude without temperature correction
516  float ampMipNoTempNoSat = 0.;
517 
518  if (_mipCutVariable == 0)
519  {
520  const LCObjectVec &amplVec = _navigator->getRelatedToObjects(hcalVec[iHits]);
521 
522  if (amplVec.size() > 0)
523  {

525  AhcAmplitude *ahcAmpl = new AhcAmplitude(obj);
526  ampMipNoTempNoSat = ahcAmpl->getAmplNOTTemperatureCorrMIP();
527  delete ahcAmpl;
528  }
529  }
530 
531  /* get the amplitude after temperature correction & saturated corrected*/
532  float ampMipFinal = hcalVec[iHits]->getEnergy();
533

536  if (_mipCutVariable == 1)

541  {
542  mipCut = ampMipNoTempNoSat;

547  if (isBadHit == false) selectedHcalHitsVec.push_back(hcalVec[iHits]);
548  }
549

552  << endl;

554 
555 
556 
557  return selectedHcalHitsVec;

564  void AngleTrackFinder::findTrackClusters(std::vector<CalorimeterHit*> &hcalVec, EVENT::ClusterVec &clusterVec)
565  {
566  streamlog_out(DEBUG4) << "\n--------------- Start to find track clusters" << endl;
567 
568  clusterVec.clear();
569 
570  LCTypedVector<CalorimeterHit>::iterator seedIter = hcalVec.begin();
571  while (seedIter != hcalVec.end())
572  {
573  CalorimeterHit *seedHit = (*seedIter);
574  const int cellID = (*seedIter)->getCellID0();
575  /*erase seed hit*/
576  hcalVec.erase(seedIter);
577 
578  /*use only hits in fine modules to find tracks*/
579  const int K = _mapper->getDecoder()->getKFromCellID(cellID);
580 
581  if (K <= 30)
582  {
583  ClusterImpl *cluster = new ClusterImpl;
584  /*first member of the cluster is the seed hit*/
585  cluster->addHit(seedHit, 1.);
586 
587  LCTypedVector<CalorimeterHit>::iterator potentialTrackHitIter = hcalVec.begin();
588  while (potentialTrackHitIter != hcalVec.end())
589  {
590  float differenceInX = fabs((*potentialTrackHitIter)->getPosition()[0] - seedHit->getPosition()[0]);
591  float differenceInY = fabs((*potentialTrackHitIter)->getPosition()[1] - seedHit->getPosition()[1]);
592 
593  CellDescription* seedHitCellDescription = _cellDescriptions->getByCellID(cellID);
594  const float seedHitTileSize = seedHitCellDescription->getSizeX();
595  const float potentialTrackHitTileSize = _cellDescriptions->getByCellID((*potentialTrackHitIter)->getCellID0())->getSizeX();
596 
597  float meanCellSize = 0.5 * (potentialTrackHitTileSize + seedHitTileSize);
598 
599  if (differenceInX < meanCellSize && differenceInY < meanCellSize)
600  {
601  streamlog_out(DEBUG4) << " New track hit found " << endl;
602 
603  /*add hit to the track cluster*/
604  cluster->addHit((*potentialTrackHitIter), 1.);
605 
606  /*erase track hit from the HCAL seeds vector*/
607  potentialTrackHitIter = hcalVec.erase(potentialTrackHitIter);
608  }
609  else
610  {
611  ++potentialTrackHitIter;
612  }
613  }/*------------- end loop over potential track hits -------------------*/
614 
615  clusterVec.push_back(cluster);
616 
617  }/*----------------- end loop over fine layers ----------------------------*/
618 
619  seedIter = hcalVec.begin();
620  }/*--------------------- end loop over AHCAL seed hits ------------------------*/
621 
622 
623  streamlog_out(DEBUG)<<"\n findTrackClusters: found "<<clusterVec.size()<<" track clusters"<<endl;
624  }
