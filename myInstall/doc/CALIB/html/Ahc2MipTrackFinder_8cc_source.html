566 int Ahc2MipTrackFinder::selectTowerNHits(TowerHitVectorHCAL &towers, unsigned int MinNHits)
567 {
568  int nTowersLeft = 0;
569  streamlog_out(DEBUG0)<<"Select HCAL Towers" <<endl;
570  TowerHitVectorHCAL::iterator itTowerI;
571 
572  for(itTowerI = towers.begin(); itTowerI != towers.end();)
573  {
574  map<float, vector< HcalSingleHit > > &thisTowerI = itTowerI->second;
575  for(map<float, vector< HcalSingleHit > >::iterator itTowerJ = thisTowerI.begin(); itTowerJ != thisTowerI.end();)
576  {
577  vector< HcalSingleHit > &HCALHits = itTowerJ->second;
578  if (HCALHits.size() >= MinNHits){ //if the currently viewed tower has enough hits, qualify as potential track
579  // and keep it
580  ++itTowerJ;
581  nTowersLeft++;
582  } else {
583  // delete current tower, but increment iterator first
584  thisTowerI.erase(itTowerJ++);
585  }
586  }
587  if(thisTowerI.size() == 0){ // if current tower is now empty
588  towers.erase(itTowerI++); // delete current tower, but increment iterator first
589  } else { // else keep it
590  ++itTowerI;
591  }
592  }
593  return nTowersLeft;
594 }
