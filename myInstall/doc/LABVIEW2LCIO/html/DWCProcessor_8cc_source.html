116  }

120  }
121 
122  int totEvent=0, matched=0;
123  int match_index=1;
124 
125  if (load_bif_tdc() != 1 || load_ahcal_tdc() != 1) {
126  streamlog_out(ERROR) << "Cannot read raw file" << endl;
127  throw marlin::StopProcessingException(this);
128  }
129  TS_diff = _BIFTDC - _AHCALTDC;
130  _AHCALTDC += TS_diff;
131 
132  while (-1) {
133 
134  if (_BIFTDC < _AHCALTDC-1) {
135  _BIFTime.push_back(_BIFTDC);
136  totEvent++;
137  if (load_bif_tdc() != 1) break;
138  } else if (_AHCALTDC < _BIFTDC-1) {

143  } else {
144  _BIFTime.push_back(_BIFTDC);
145  totEvent++;
146  matched++;
147  if (load_bif_tdc() != 1) break;
148  if (load_ahcal_tdc() != 1) break;
149  _AHCALTDC += TS_diff;
150  }
151 
152  if (totEvent == 20 && matched <= 10) {
153  match_index++;
154  fseek(_bifFile, 0, SEEK_SET);
155  fseek(_ahcalFile, 0, SEEK_SET);
156  load_bif_tdc();
157  load_ahcal_tdc();
158  for (i=0; i<match_index/2; i++) {
159  if (match_index%2) load_bif_tdc();
160  else load_ahcal_tdc();
161  }
162  totEvent=0;
163  matched=0;
164  TS_diff = _BIFTDC - _AHCALTDC;
165  _AHCALTDC += TS_diff;
166  _BIFTime.clear();
167  }
168  }
169 
170  fclose(_ahcalFile);
171  fclose(_bifFile);
172 
173  _index=0;
174 
175  _dwcTree = (TTree *)_dwcFile->Get("DelayWireChambers");
176  int nEvent = _dwcTree->GetEntries();
177  long long int dwc_time;
178  char branchName[16];
179  for (i=0;i<16;i++) {
180  sprintf(branchName, "dwc_hits_ch%d",i);
181  _dwcTree->SetBranchAddress(branchName,&_dwcCh[i]);
182  }
183  _dwcTree->SetBranchAddress("timeSinceStart",&dwc_time);
184 
185  for (i=0;i<nEvent;i++) {
186  _dwcTree->GetEntry(i);
187  _DWCTime.push_back(dwc_time);
188  _DWCTrig.push_back(i);
189  }
190 
191  event_match();
192  printParameters();

330  int DWCProcessor::load_ahcal_tdc() {
331  unsigned long long int TS = 0;
332  unsigned char minibuf[8];
333  int length, type;
334  while (1) {
335  if (fread(minibuf, 1, 1, _ahcalFile) <= 0) return -1;
336  if (minibuf[0] != 0xCD) continue;
337  if (fread(minibuf, 1, 1, _ahcalFile) <= 0) return -1;
338  if (minibuf[0] != 0xCD) continue;
339  if (fread(minibuf, sizeof(minibuf), 1, _ahcalFile) <= 0) return -1;
340  length = (int) minibuf[0] + ((int) minibuf[1] << 8);
341  if ((minibuf[0] != 0x10) || (minibuf[1] != 0x0) || (minibuf[7] != 0x08)) {
342  //not packets we want
343  fseek(_ahcalFile, length, SEEK_CUR);
344  continue;
345  }
346  if (fread(minibuf, sizeof(minibuf), 1, _ahcalFile) <= 0) return -1;
347  if ((minibuf[0] != 0x45) || (minibuf[1] != 0x4D) || (minibuf[2] != 0x49) || (minibuf[3] != 0x54)) {
348  fseek(_ahcalFile, 8, SEEK_CUR);
349  continue;
350  }
351  type = minibuf[4];
352 
353  if (fread(minibuf, sizeof(minibuf), 1, _ahcalFile) <= 0) return -1;
354  if ((minibuf[6] != 0xAB) || (minibuf[7] != 0xAB)) continue;
355  TS = (unsigned long long int) minibuf[0] +
356  ((unsigned long long int) minibuf[1] << 8) +
357  ((unsigned long long int) minibuf[2] << 16) +
358  ((unsigned long long int) minibuf[3] << 24) +
359  ((unsigned long long int) minibuf[4] << 32) +
360  ((unsigned long long int) minibuf[5] << 40);
361 
362  if (type != 0x10) continue;
363 
364  _AHCALTDC = TS;
365  return 1;
366  }

369  int DWCProcessor::load_bif_tdc() {
370  unsigned long long int time_h = 0, time_l = 0;
371  unsigned long long int time = 0;
372  unsigned char minibuf[8];
373 
374  while (-1) {
375  if (fread(minibuf, sizeof(minibuf), 1, _bifFile) <= 0) return -1;
376  //Bug with gcc for strict aliasing rules
377  time_h = *((u_int32_t *) minibuf + 1) & 0x0000FFFF;
378  time_l = *((u_int32_t *) minibuf);
379 
380  time = ((u_int64_t) time_h << 32) | time_l;
381  switch (minibuf[7] >> 4) {
382  case 0:
383  case 1:
384  if (fread(minibuf, sizeof(minibuf), 1, _bifFile) <= 0) return -1;
385  _BIFTDC = time;
386  return 1;
387  case 2:
388  case 3:
389  break;
390  default:
391  fprintf(stderr, "unknown BIF packet\n");
392  return 0;
393  }
394  }

397  void DWCProcessor::event_match() {
398  long long int bifTime_norm;
399  long long int dwcTime_norm;
400  long long int bifTime_prev;
401  long long int dwcTime_prev;
402  int refIndex = 0;
403  unsigned int i;
404 
405  for (i=1; i<_BIFTime.size() && i<_DWCTime.size(); i++) {
406  if ((_BIFTime[i]-_BIFTime[i-1])*BIF_US > BUNCH_THR) {
407  bifTime_norm = (_BIFTime[i]-_BIFTime[refIndex])*BIF_US;
408  dwcTime_norm = _DWCTime[i]-_DWCTime[refIndex];
409  bifTime_prev = (_BIFTime[i-1]-_BIFTime[refIndex])*BIF_US;
410  dwcTime_prev = _DWCTime[i-1]-_DWCTime[refIndex];
411 
412  if (abs(bifTime_norm-dwcTime_norm) > abs(bifTime_prev-dwcTime_norm)) {
413  cout<<"BIF trigger missing"<<endl;
414  cout<<" bifTime[i]:"<<_BIFTime[i]
415  <<", bifTime[ref]:"<<_BIFTime[refIndex]<<endl;
416  cout<<" bifTime:"<<bifTime_norm
417  <<", dwcTime:"<<dwcTime_norm
418  <<", bifTime_prev:"<<bifTime_prev<<endl;
419  cout<<"removing "<<i-refIndex<<" events from "<<refIndex<<" to "<<i-1<<endl;
420  _BIFTime.erase(_BIFTime.begin()+refIndex,_BIFTime.begin()+i);
421  _DWCTime.erase(_DWCTime.begin()+refIndex,_DWCTime.begin()+i+1);
422  _DWCTrig.erase(_DWCTrig.begin()+refIndex,_DWCTrig.begin()+i+1);
423  i=refIndex;
424  } else if (abs(bifTime_norm-dwcTime_norm) > abs(bifTime_norm-dwcTime_prev)) {
425  cout<<"DWC trigger missing"<<endl;
426  cout<<" bifTime:"<<bifTime_norm
427  <<", dwcTime:"<<dwcTime_norm
428  <<", dwcTime_prev:"<<dwcTime_prev<<endl;
429  cout<<"removing "<<i-refIndex<<" events from "<<refIndex<<" to "<<i-1<<endl;
430  _BIFTime.erase(_BIFTime.begin()+refIndex,_BIFTime.begin()+i);
431  _DWCTime.erase(_DWCTime.begin()+refIndex,_DWCTime.begin()+i-1);
432  _DWCTrig.erase(_DWCTrig.begin()+refIndex,_DWCTrig.begin()+i-1);
