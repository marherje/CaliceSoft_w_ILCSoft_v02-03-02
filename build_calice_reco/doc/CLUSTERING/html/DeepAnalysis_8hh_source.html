395  void inertia_3d(float *wr){
396  float a[3][3], zr[3][3], work[20];
397  memset(wr, 0, sizeof(wr));
398  memset(a, 0, sizeof(a));
399  memset(zr, 0, sizeof(zr));
400  float weig = 0.0;
401 
402  // Calculate tensor of inertia
403  for(RSIterator<Hit> hitIterator(this); hitIterator.next();){
404  Hit &hit = *hitIterator;
405  //Point3D p=h.cartesianPoint - s;
406  Point3D p(hit.cartesianPoint.x - centerOfGravity.x,
407  hit.cartesianPoint.y - centerOfGravity.y,
408  hit.cartesianPoint.z - centerOfGravity.z);
409 
410  a[0][0] += hit.amplGeV * (p.y * p.y + p.z * p.z);
411  a[1][1] += hit.amplGeV * (p.x * p.x + p.z * p.z);
412  a[2][2] += hit.amplGeV * (p.x * p.x + p.y * p.y);
413  a[0][1] -= hit.amplGeV * (p.x * p.y);
414  a[0][2] -= hit.amplGeV * (p.x * p.z);
415  a[1][2] -= hit.amplGeV * (p.y * p.z);
416  weig += hit.amplGeV;
417  }
418  a[1][0] = a[0][1];
419  a[2][0] = a[0][2];
420  a[2][1] = a[1][2];
421 
422  int ierror = 0;
423  // Eigenvalue problem solve
424  EISRS1(3, 3,(float *)a, wr, (float *)zr, ierror, work);
425  if(ierror) cerr<<" ERROR in INERTIA"<<endl;
426 
427  clusterAxisVector.calc(zr[0][0], zr[1][0], zr[2][0]);
428  double cdir = 1.;
429  // Correct direction --- should go from the center of coordinates
430  Vector3D vend = centerOfGravity + clusterAxisVector;
431  if(vend.module() - clusterRadius < 0.){
432  clusterAxisVector.calc(-clusterAxisVector.x, -clusterAxisVector.y, -clusterAxisVector.z);
433  cdir = -1.;
434  }
435 
436  for(int i = 0; i < 3 ; i++){
437  v1[i] = cdir * zr[i][0];
438  v2[i] = cdir * zr[i][1];
439  v3[i] = cdir * zr[i][2];
440  }
441 
442  double DIS1 = 0., DIS2 = 0., DIS3 = 0., weig2 = 0.;
443 
444  for(RSIterator<Hit> hitIterator(this); hitIterator.next();){
445  Hit &hit = *hitIterator;
446  weig2 += hit.amplGeV * hit.amplGeV;
447  double tmp = dist2line(hit.cartesianPoint, centerOfGravity, clusterAxisVector);
448 
449  DIS1 += (tmp * hit.amplGeV) * (tmp * hit.amplGeV);
450  DIS2 += fabs(tmp) * hit.amplGeV;
451  DIS3 += fabs(tmp);
452  }
453  chi[0] = sqrt(DIS1/weig2);
454  chi[1] = DIS2/weig;
455  chi[2] = DIS3/getNumberOf<Hit>();
456  }

460  void calc_prop(DeepAnalysis &rs, bool for_em=false){
461  calc_stat();
462  calc_color(rs);
463  clusterRadius = centerOfGravity.module();
464  clusterAxisVector.calc(0.,0.,0.);
465  clusterLongestRadius = r2 = r3 = forwRadius = backwRadius = ecc = chi[0] = chi[1] = chi[2] = 0.;
466  begn = last = 0;
467 
468  unsigned nnn = getNumberOf<Hit>();
469  if(nnn > 2){ // cluster could be treated as inertia tensor
470  float wr[3]; // Inertia tensor EigenValues
471  inertia_3d(wr);
472 
473  //Ellipsoid parameter calculations
474  for(unsigned i = 0; i < 3; i++) {
475  wr[i] = sqrt(wr[i]/energySum);//Normalize inertia tensor eigen values
476  }
477 
478  double chisq = chi[0];
479  clusterLongestRadius = sqrt(wr[1]*wr[2]);
480  r2 = sqrt(wr[0] * wr[2]); // spatial axis length
481  r3 = sqrt(wr[0] * wr[1]);
482  r2 = r3 = max(max(r2, r3), 0.5 * rs.detector.cell_size);
483  double r_hit_max = -10000.;
484  double d_begn = 100000.;
485  double d_last = -100000.;
486 
487  for(RSIterator<Hit> hitIterator(this); hitIterator.next();){
488  //Vector3D d=hi->cartesianPoint-s;
489  Vector3D d(hitIterator->cartesianPoint.x - centerOfGravity.x,
490  hitIterator->cartesianPoint.y - centerOfGravity.y,
491  hitIterator->cartesianPoint.z - centerOfGravity.z);
492 
493  double r_max = d.module();
494  if(r_max > r_hit_max) r_hit_max = r_max;
495 
496  float proj = d.dot(clusterAxisVector);
497  if(proj < d_begn){
498  d_begn = proj;
499  begn = &hitIterator;
500  }
501  if(proj > d_last){
502  d_last = proj;
503  last = &hitIterator;
504  }
505  }
506  forwRadius = fabs(d_last);
507  backwRadius = fabs(d_begn);
508  // Calculate other parameters of ellipsoid
509  volume = 4.* M_PI * clusterLongestRadius * r2 * r3/3.; // ellipsoid volume
510  ellipAverageRadius = pow(volume,0.333333); // average radius
511  density = energySum/volume; // density [GeV]/[cm]
512 
513  float r_min_ave;
514  r_min_ave = sqrt(r2 * r3/2.0);
515  if(r_min_ave < 0.001) r_min_ave = chisq;
516  if(r_min_ave < 0.001) r_min_ave = 0.5*rs.detector.cell_size;
517 
518  clusterLongestRadius = max(max(forwRadius, backwRadius), clusterLongestRadius);
519  ecc = r_min_ave/clusterLongestRadius; // Cluster Eccentricity
520 
521  if(ecc > 1.0)
522  cout<<" BUG ::: Eccentricity > 1 ; Chisq = "<<chi[0]
523  <<" : "<<clusterLongestRadius
524  <<" : "<<r_min_ave
525  <<" : "<<ecc
526  <<endl;
527 
528  } else if(nnn == 2){ // 2-pt cluster parameters line in space
529  RSIterator<Hit> hitIterator(this);
530  hitIterator.next();
531  Hit &first = *hitIterator;
532  hitIterator.next();
533  Hit &second = *hitIterator;
534 
535  Vector3D vect(second.cartesianPoint.x - first.cartesianPoint.x,
536  second.cartesianPoint.y - first.cartesianPoint.y,
537  second.cartesianPoint.z - first.cartesianPoint.z);
538 
539  clusterAxisVector.calc(directionCos(vect));
540 
541 
542  Vector3D vend = centerOfGravity + clusterAxisVector;
543  // Correct direction --- should go from the center of coordinates
544  if(vend.module() - clusterRadius < 0.)
545  clusterAxisVector.calc(-clusterAxisVector.x, -clusterAxisVector.y, -clusterAxisVector.z);
546  volume = sqrt(vect.length2()) * rs.detector.cell_size * rs.detector.sampling;
547  density = energySum/volume; // density [GeV]/[cm]
548 
549  // Find first and last points of cluster
550  Vector3D diffVec_first(first.cartesianPoint.x - centerOfGravity.x,
551  first.cartesianPoint.y - centerOfGravity.y,
552  first.cartesianPoint.z - centerOfGravity.z);
553 
554  backwRadius = clusterAxisVector.dot(diffVec_first);
555 
556  Vector3D diffVec_second(second.cartesianPoint.x - centerOfGravity.x,
557  second.cartesianPoint.y - centerOfGravity.y,
558  second.cartesianPoint.z - centerOfGravity.z);
559 
560  backwRadius = clusterAxisVector.dot(diffVec_second);
561 
562  begn = &first;
563  last = &second;
564 
565  if(backwRadius > forwRadius){
566  double tmp = backwRadius;
567  backwRadius = forwRadius;
568  forwRadius = tmp;
569  last = &first;
570  begn = &second;
571  last = &first;
572  }
573 
574  chi[0] = chi[1] = chi[2] = 0.5 * rs.detector.cell_size;
575 
576  } else if(nnn == 1){ // 1-pt cluster parameters point in space
577  // vol = length * tile_area * sampling (latter to do)
578  begn = last = getChild<Hit>();
579  } else
580  cerr<<"Empty cluster?"<<endl;
581  }
