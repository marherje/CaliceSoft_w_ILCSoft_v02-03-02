300  // ampl_gev = the energy amplitude of one hit, in GeV
301  // ampl_mip = the energy amplitude of one hit, in number of MIPs
302  // type = a pre-assigned hit type (according to the energy)
303  // layer = the layer in which the hitted cell is found
304  deepAnalysis.add_hit(position.at(0), //x
305  position.at(1), //y
306  position.at(2), //z

351 #endif

355  double trk_engySum = 0;

358  cout<<"trk_ncl="<<trk_ncl<<" trk_nhit="<<trk_nhit<<" trk_engySum="<<trk_engySum<<endl;

508  void DeepAnalysisProcessor::updateCorrelations() {
509 
510  int populatedLayer = 0;
511 
512  std::vector<std::vector<unsigned> > layerOccupation;
513  layerOccupation.resize(_inputCollectionNames.size());
514  _gevPerMip.resize(_inputCollectionNames.size());
515  _populatedLayerIndex.resize(_inputCollectionNames.size());
516 
517  for (unsigned detectorIndex = 0; detectorIndex < _inputCollectionNames.size(); detectorIndex++) {
518  for (unsigned moduleIndex = 0; moduleIndex < _mapping[detectorIndex].getNModules(); moduleIndex++) {
519  unsigned layerIndex =
520  CellIndex(_mapping[detectorIndex].getGeometricalCellIndex(moduleIndex, 0)).getLayerIndex();
521 
522  if (layerIndex + 1 > layerOccupation[detectorIndex].size()) {
523  layerOccupation[detectorIndex].resize(layerIndex+1);
524  }
525 #ifdef HCALRECO_DEBUG
526  std::cout << "DeepAnalysisProcessor::updateCorrelations: found layer "
527  << layerIndex << " in detector " << detectorIndex << std::endl;
528 #endif
529  (layerOccupation[detectorIndex])[layerIndex]++;
530  }
531 
532  _populatedLayerIndex[detectorIndex].clear();
533  _gevPerMip[detectorIndex].clear();
534  for (unsigned layer = 0; layer < layerOccupation[detectorIndex].size(); layer++) {
535  if (!(layerOccupation[detectorIndex])[layer]) {
536  _populatedLayerIndex[detectorIndex].push_back(-1);
537  _gevPerMip[detectorIndex].push_back(0.);
538  }
539  else {
540  _populatedLayerIndex[detectorIndex].push_back(populatedLayer);
541  int i;
542  for (i = layer-1; i >= 0; i--) {
543  if ((_populatedLayerIndex[detectorIndex])[i]>=0) break;
544  }
545  if (layer==0)
546  _gevPerMip[detectorIndex].push_back(_samplingFractionA[detectorIndex] * _samplingFactor);
547  else
548  _gevPerMip[detectorIndex].push_back(_samplingFractionA[detectorIndex]*
549  (1+_samplingFractionB[detectorIndex]*(layer-i-1))
550  *_samplingFactor);
551  populatedLayer++;
552  }
553  }
554 #ifdef HCALRECO_DEBUG
555  std::cout << "DeepAnalysisProcessor::updateCorrelations: results for detector "
556  << detectorIndex << " :" << std::endl;
557 
558  for (int layer = 0; layer < _populatedLayerIndex[detectorIndex].size(); layer++) {
559  if ((_populatedLayerIndex[detectorIndex])[layer] >= 0) {
560  std::cout << " " << layer << ": " << (_populatedLayerIndex[detectorIndex])[layer]
561  << "/" << (_gevPerMip[detectorIndex])[layer] << std::endl;
562  }
563  }
564  std::cout << std::endl;
565 #endif
566  }
567  }
