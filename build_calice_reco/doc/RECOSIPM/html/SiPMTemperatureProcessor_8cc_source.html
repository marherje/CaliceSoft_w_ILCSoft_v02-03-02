196  void SiPMTemperatureProcessor::processEvent(LCEvent *evt)

198  streamlog_out(DEBUG0)<<"\n\n Event "<<evt->getEventNumber()<<std::endl;

207

218 
219  streamlog_out(DEBUG)<<"\n maxModule: "<< _mapper->getMaxModule()<<" maxChip: "<<_mapper->getMaxChip()<<" maxChannel: "<<_mapper->getMaxChannel()<<endl;
220  /*fill the temperature container if something changed in the temperature provider*/
221  if (_ahcSroModDataColChanged == true || _temperatureSensorCalibrationColChanged == true)
222  {
223  _colTemperature = new LCCollectionVec(LCIO::LCGENERICOBJECT);
224 
225  /*yes, the AHCAL modules start from 1, because _mapper->getMaxModule() gives either 31 or 39*/
226  for (unsigned int module = 1; module < _mapper->getMaxModule(); ++module)
227  {
228  for (unsigned int chip = 0; chip < _mapper->getMaxChip(); ++chip)
229  {
230  for (unsigned int channel = 0; channel < _mapper->getMaxChannel(); ++channel)
231  {
232  const float temperature = _tempProvider->getCellTemp(module, chip, channel);
233  const float temperatureError = _tempProvider->getCellTempError(module, chip, channel);
234 
235  streamlog_out(DEBUG)<<" mod/chip/chan="<<module<<"/"<<chip<<"/"<<channel<<" temperature="<<temperature<<" +- "<<temperatureError<<endl;
236  try {
237  int moduleID = _mapper->getDecoder()->getModuleID(module, chip, channel);
238 
239  SimpleValue *val = new SimpleValue(moduleID, temperature, temperatureError, 1);
240  _colTemperature->addElement(val);
241 
242  streamlog_out(DEBUG0)<<" moduleID="<<moduleID<<std::endl;
243  }
244  catch(BadDataException& e)
245  {
246  streamlog_out(DEBUG0) << " invalid cell id for module " << module <<", chip "<<chip
247  <<" and channel "<<channel<<" \n"<< e.what() << std::endl;
248  }
249 
250  }
251  }
252  }/*end loop over module*/
253 
254  const std::string encoding = _mapper->getDecoder()->getCellIDEncoding();
255  LCParameters &theParam = _colTemperature->parameters();

261  }
262 
263  /*----------------------------------------------------------------------*/

265  _ahcSroModDataColChanged = false;
266  if (_temperatureSensorCalibrationColChanged == true)
267  _temperatureSensorCalibrationColChanged = false;
