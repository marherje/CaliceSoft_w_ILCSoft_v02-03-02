454  void SiPMCalibrateProcessor::calibrateEnergyAndFillOutputCollections(SiPMCalibrations* calibration,
455  const float rawEnergy,
456  LCCollection* ahcHitOutputCol,
457  LCCollection* ahcAmplOutputCol,
458  LCRelationNavigator &relationNavigator)
459  {
460  const SiPMCalibrationStatusBits bits = SiPMCalibrationStatusBits( calibration->getStatus() );
461  streamlog_out(DEBUG0) << " STATUS bits: " << bits << endl;
462 
463  if ( _filterDeadCells && bits.isDead() )
464  {
465  streamlog_out(DEBUG0) << " SKIP -- channel is marked as dead: " << endl;
466  return;
467  }
468 
469  if ( _filterDefaultCells && bits.hasDefault() )
470  {
471  streamlog_out(DEBUG0) << " SKIP -- channel has default calibration: " << endl;
472  return;
473  }
474 
475  /*-------------------- calibration coefficients --------------*/
476  LinearFitCompound *lfcMIP = calibration->getMIP();
477  LinearFitCompound *lfcGain = calibration->getGain();
478  SimpleValue *interCalibContainer = calibration->getInterCalibration();
479  SatCorrFunction *saturationContainer = calibration->getSaturationCorrection();
480 
481  /*------------------------------------------------
482  get pedestal
483  -----------------------------------------------*/
484  float pedestal = 0;
485  float pedestalError = 0;
486 
487  SimpleValue *pedestalContainer = calibration->getPedestal();
488  if (_pedestalSubtraction == 1)
489  {
490  if (pedestalContainer == NULL) {
491  streamlog_out(DEBUG0) << " SKIP -- no pedestal available: " << endl;
492  return;
493  }
494 
495  pedestal = pedestalContainer->getValue();
496 
497  /*
498  What about the significance cut?
499  need pedestal width !
500  */
501  }
502 
503 
504  /*----------------------------------------------
505  get temperature
506  -----------------------------------------------*/
507  SimpleValue *temperatureObject = calibration->getTemperature();
508  float temperature = 0;
509  if (temperatureObject != NULL) temperature = temperatureObject->getValue();
510  else
511  {
512  streamlog_out(ERROR)<<" No temperature found"<<endl;
513  exit(1);
514  }
515 
516  int cellID = calibration->getCellID();
517  _mapper->getDecoder()->setCellIDEncoding(calibration->getCellIDEncoding());
518 
519  streamlog_out(DEBUG0) << " cellID: "<<cellID<<endl;
520  streamlog_out(DEBUG0) << " from module " << _mapper->getModuleFromCellID(cellID) << endl;
521  streamlog_out(DEBUG0) << " with amplitude: "<< rawEnergy << endl
522  << " @ temperature: " << temperature << " deg. C" << endl;
523 
524  /*------------------------------------------------
525  get MIP, gain and IC
526  -----------------------------------------------*/
527 
528  /*---------------------- MIP ---------------------------------*/
529  float mipValue = 0;
530 
531  if (_doMipTempCorr) mipValue = lfcMIP->eval(temperature);
532  else
533  mipValue = lfcMIP->getConstant();
534  if (mipValue <= 0)
535  {
536  /*
537  streamlog_out(ERROR0) << " ERROR - ignoring cell with negative/"
538  << "zero MIP coefficient at I/J/K "
539  << _mapper->getDecoder()->getIFromCellID(cellID)
540  <<"/"<<_mapper->getDecoder()->getJFromCellID(cellID)
541  <<"/"<<_mapper->getDecoder()->getKFromCellID(cellID)
542  << endl;
543  */
544  return;
545  }
546 
547 
548  /*---------------------- gain ---------------------------------*/
549  float gainValue = 0;
550  if (_doGainTempCorr) gainValue = lfcGain->eval(temperature);
551  else gainValue = lfcGain->getConstant();
552  if (gainValue <= 0)
553  {
554  /*
555  streamlog_out(ERROR0) << " ERROR - ignoring cell with negative/"
556  << "zero gain coefficient at I/J/K "
557  << _mapper->getDecoder()->getIFromCellID(cellID)
558  <<"/"<<_mapper->getDecoder()->getJFromCellID(cellID)
559  <<"/"<<_mapper->getDecoder()->getKFromCellID(cellID)
560  << endl;
561  */
562  return;
563  }
564 
565  float interCalibrationValue = interCalibContainer->getValue();
566  if (interCalibrationValue <= 0)
567  {
568  streamlog_out(ERROR0) << " ERROR - ignoring cell with negative/"
569  << "zero inter calibration at I/J/K "
570  << _mapper->getDecoder()->getIFromCellID(cellID)
571  <<"/"<<_mapper->getDecoder()->getJFromCellID(cellID)
572  <<"/"<<_mapper->getDecoder()->getKFromCellID(cellID)
573  << endl;
574  return;
575  }
576 
577  streamlog_out(DEBUG0) << " - pedestal: " << pedestal << endl;
578  streamlog_out(DEBUG0) << " - MIP: " << mipValue << endl;
579  streamlog_out(DEBUG0) << " - gain: " << gainValue << endl;
580  streamlog_out(DEBUG0) << " - intercalibration: " << interCalibrationValue << endl;
581 
582 
583  //check if gainConstant or IC is default => put gainConst to have fixed LY.
584  //This prevents an event to have the "strange" desaturated amplitudes
585  if ( _correctDefaultGainToLY && (bits.gainConstantIsDefault() || bits.interCalibrationIsDefault()) ) {
586  gainValue = mipValue * interCalibrationValue / _fixedLY;
587  }
588 
589  /*----------------------------------------------
590  do calibration
591  ---------------------------------------------*/
592  float calibratedEnergy = 0;
593 
594  if ( _doSaturationCorr )
595  {
596  streamlog_out(DEBUG0)<< " - amplitude [pix]: " << rawEnergy * mipValue/gainValue * interCalibrationValue<<endl;
597  streamlog_out(DEBUG0)<< " - corrected [pix]: "
598  << saturationContainer->deSaturate( (rawEnergy - pedestal)
599  * interCalibrationValue / gainValue )
600  <<endl;
601 
602  calibratedEnergy = saturationContainer->deSaturate( (rawEnergy - pedestal) * interCalibrationValue / gainValue )
603  * gainValue / interCalibrationValue / mipValue;
604 
605  streamlog_out(DEBUG0) << " - saturationCorrectionFactor: "
606  << saturationContainer->deSaturate( (rawEnergy - pedestal) * interCalibrationValue / gainValue )
607  / ( (rawEnergy - pedestal) * interCalibrationValue / gainValue)
608  << endl;
609  }
610  else
611  {
612  calibratedEnergy = (rawEnergy - pedestal) / mipValue;
613  }
614 
615 
616  streamlog_out(DEBUG0)<<" - calibratedEnergy[MIPs]: "<<calibratedEnergy<<endl;
617  /*--------------------------------------
618  * if requested, do error calculation
619  ---------------------------------------*/
620  float calibratedEnergyError = 0;
621  if ( _doErrorCalculation )
622  {
623  if (_pedestalSubtraction == 1) pedestalError = pedestalContainer->getError();
624 
625  const float interCalibrationError = interCalibContainer->getError();;
626  const float temperatureError = temperatureObject->getError();
627 
628  float mipError = 0;
629  float gainError = 0;
630  if (_doMipTempCorr) mipError = lfcMIP->evalErr(temperature,temperatureError);
631  else mipError = lfcMIP->getConstantError();
632 
633  if (_doGainTempCorr) gainError = lfcGain->evalErr(temperature,temperatureError);
634  else gainError = lfcGain->getConstantError();
635 
636  streamlog_out(DEBUG0) << " - MIP error: " << mipError << endl;
637  streamlog_out(DEBUG0) << " - gain error: " << gainError << endl;
638  streamlog_out(DEBUG0) << " - intercalibration error: " << interCalibrationError << endl;
639 
640  const float relativeErrorPedSubSquare =
641  ( pow(calibratedEnergyError, 2) + pow(pedestalError, 2) ) / pow( rawEnergy - pedestal, 2);
642 
643  /*----------------------------------------
644  if requested, do saturation correction
645  ---------------------------------------*/
646  if ( _doSaturationCorr )
647  {
648  const float saturatedSignalInPixel = (rawEnergy - pedestal) * interCalibrationValue / gainValue ;
649  const float unsaturatedSignalInPixel = saturationContainer->deSaturate( saturatedSignalInPixel );
650 
651  /* assuming that the intercalibration error cancels in first order*/
652  const float saturatedSignalInPixelError =
653  sqrt( relativeErrorPedSubSquare + pow(gainError/gainValue,2) ) * saturatedSignalInPixel;
654 
655  calibratedEnergyError =
656  sqrt( pow(saturationContainer->deSaturatedError(saturatedSignalInPixel,
657  saturatedSignalInPixelError)/unsaturatedSignalInPixel,2)
658  + pow(mipError/mipValue,2) ) * calibratedEnergy;
659  }
660  else
661  {
662  calibratedEnergyError =
663  sqrt( relativeErrorPedSubSquare + pow(mipError/mipValue,2) ) * calibratedEnergy;
664  }
665  }
666 
667  /*-----------------------------------------------
668  0.5 MIP cut
669  ----------------------------------------------*/
670  if ( _zeroSuppression && calibratedEnergy < _mipCut )
671  {
672  streamlog_out(DEBUG0) << " STOP - " << calibratedEnergy
673  << " - below threshold of: " << _mipCut << endl;
674  streamlog_out(DEBUG0)<<" ------------------------------"<<endl;
675  return;
676  }
677 
678  streamlog_out(DEBUG0)<< " DONE - calibrated amplitude: " << calibratedEnergy;
679  if (_doErrorCalculation) streamlog_out( DEBUG0 ) << " +- " << calibratedEnergyError;
680  streamlog_out(DEBUG0)<< endl;
681 
682 
683 
684  /*---------------------------------------------
685  write output calorimeter hit
686  --------------------------------------------*/
687  CalorimeterHitImpl *newHit = new CalorimeterHitImpl();
688  if (_scaleEnergy)
689  {
690  newHit->setEnergy( calibratedEnergy*_energyScaleFactor );
691  newHit->setEnergyError(calibratedEnergyError*_energyScaleFactor );
692  }
693  else
694  {
695  newHit->setEnergy(calibratedEnergy);
696  newHit->setEnergyError(calibratedEnergyError);
697  }
698  newHit->setCellID0(cellID);
699 
700  _cellDescriptions->getDecoder()->setCellIDEncoding(calibration->getCellIDEncoding());
701  CellDescription* cellDescription = _cellDescriptions->getByCellID(cellID);
702  float newPosition[3] = {cellDescription->getX(), cellDescription->getY(), cellDescription->getZ()};
703 
704  streamlog_out(DEBUG0)<<" position: ("<<newPosition[0]<<", "<<newPosition[1]<<", "<<newPosition[2]<<std::endl;
705 
706  newHit->setPosition(newPosition);
707 
708  /*add collection of AHCal calibrated hits to the event*/
709  ahcHitOutputCol->addElement(newHit);
710 
711  streamlog_out(DEBUG0)<<" \n output collection "<<_ahcHitOutputColName<<" has "
712  <<ahcHitOutputCol->getNumberOfElements()<<" elements"<<endl;
713  streamlog_out(DEBUG0)<<"=====================\n\n" << endl;
714 
715  /*----------------------------------------------------------------
716  for the relation between CalorimeterHits and SimCalorimeterHits
717  ------------------------------------------------------------------*/
718  if (_createSimRecRelation == true)
719  _recoContainer->fillByCellID(newHit->getCellID0(), newHit);
720 
721  /*--------------------------------------------------
722  write output AHCal hit amplitude in different units
723  -------------------------------------------------*/
724  /*set the amplitude in the different units*/
725  AhcAmplitude *ahcAmplitude = new AhcAmplitude(cellID,
726  rawEnergy,
727  rawEnergy - pedestal,
728  (rawEnergy - pedestal)/lfcMIP->eval(temperature),
729  (rawEnergy - pedestal)/lfcMIP->getConstant(),
730  calibratedEnergy * _mipToGeVFactor,
731  temperature);
732 
733  ahcAmplOutputCol->addElement(ahcAmplitude);
734 
735  LCParameters &paramAmpl = ahcAmplOutputCol->parameters();
736  paramAmpl.setValue("DataDescription", ahcAmplitude->getDataDescription());
737 
738  /* ---------------------------------------------------------
739  Set the relation between CalorimeterHit and AhcAmplitude
740  -----------------------------------------------------------*/
741  relationNavigator.addRelation(newHit, ahcAmplitude);
742 
743  }
