228  necessary_condition=true;

261

264

266  // << evtP->getEventNumber()

269  wafer_index_hist.histogram(layer_i)->fill(col_i,row_i);
270 # endif
271  n_squares++;
272  mark=true;
273  if (!_clusterColName.empty()) {

276  }

278  }

359  // std::cout << "EventViewProcessor::processEvent> square event:" << evtP->getRunNumber() << " " << evtP->getEventNumber() << std::endl;
360  }

363 # ifdef HAVE_CALICEGUI

367  gui->markDisplayList( evtP->getRunNumber() , evtP->getEventNumber() );
368  }
