454 void HistMgr::writeHistograms(Group_t &group, const std::string &group_name, bool snapshot)
455 {
456  if (!_writerKit || group.header().fileName().empty()) return;
457 
458  std::string folder_name;
459  // std::string file_name=group.header().fileName();
460  if (snapshot || group.header().version()>0) {
461  std::stringstream folder_name_with_version;
462  if (group.header().folderName().empty()){
463  if (snapshot) {
464  folder_name_with_version << "snapshot_" << group.header().version();
465  }
466  }
467  else {
468  folder_name_with_version << group.header().folderName() << "_" << group.header().version();
469  }
470  folder_name=folder_name_with_version.str();
471  }
472  else {
473  folder_name=group.header().folderName();
474  }
475 
476  HistWriter *writer=0;
477 
478  // for fast access the histogram collections are stored in a map.
479  // During histogram creating the map structure changes. Thus, it is no
480  // so easy to maintain an index of all histogram which belong to a certain
481  // group.
482  // So, the whole list of histograms is worked through and all histograms
483  // which have the requested group_id are written.
484 
485  UInt_t n_histograms=0;
486  UInt_t n_graphs=0;
487  bool first=true;
488  for (HistogramList_t::iterator col_iter=group._histogramList.begin();
489  col_iter!=group._histogramList.end();
490  col_iter++) {
491 
492  HistogramCollection_t &hist_col=*(*col_iter);
493  if (hist_col.n()==0) {
494  std::cerr << "HistMgr::writeHistograms> Histogram collection " << col_iter.name()
495  << " is empty." << std::endl;
496  continue;
497  }
498 
499 
500  // the file is only created if the group contains at least one histogram;
501  if (!writer) {
502  writer=_writerKit->createWriter(group.header().fileName());
503  if (!folder_name.empty()) {
504  writer->enterDir(folder_name,true);
505  }
506  }
507 
508  writer->enterDir(col_iter.name(),true);
509 
510  UInt_t n_hists_written=0;
511  UInt_t n_empty_hists_written=0;
512  if (hist_col.is2D()) {
513  for (UInt_t major_i=0; major_i<(UInt_t) hist_col.nMajor(); major_i++) {
514  for (UInt_t hist_i=0; hist_i<(UInt_t) hist_col.nMinor(major_i); hist_i++) {
515  const FloatHistogram1D *a_hist=hist_col.histogram(major_i,hist_i);
516  if (a_hist->entries()>0) {
517  n_hists_written++;
518  }
519  else {
520  n_empty_hists_written++;
521  }
522  std::stringstream message;
523  std::string full_name(hist_col.getName(major_i));
524  std::string::size_type base_name_end_i=full_name.find("::");
525  if (base_name_end_i==std::string::npos || base_name_end_i+2>=full_name.size()) {
526  base_name_end_i=0;
527  }
528  else {
529  base_name_end_i+=2;
530  }
531  message << full_name.substr(base_name_end_i,full_name.size()-base_name_end_i)
532  << "_"
533  << major_i
534  << "_"
535  << hist_i;
536  writer->writeToCurrentDir(*a_hist,message.str());
537  }
538  }
539  }
540  else {
541  for (UInt_t hist_i=0; hist_i<(UInt_t) hist_col.n(); hist_i++) {
542  const FloatHistogram1D *a_hist=hist_col.histogram(hist_i);
543  if (a_hist->entries()>0) {
544  n_hists_written++;
545  }
546  else {
547  n_empty_hists_written++;
548  }
549  std::stringstream message;
550  std::string full_name(hist_col.getName(hist_i));
551  std::string::size_type base_name_end_i=full_name.find("::");
552  if (base_name_end_i==std::string::npos || base_name_end_i+2>=full_name.size()) {
553  base_name_end_i=0;
554  }
555  else {
556  base_name_end_i+=2;
557  }
558  message << full_name.substr(base_name_end_i,full_name.size()-base_name_end_i)
559  << "_"
560  << hist_i;
561  writer->writeToCurrentDir(*a_hist,message.str());
562  }
563  }
564 
565  if (n_hists_written + n_empty_hists_written>0) {
566  if (first ) {
567  std::cout << "Written Histograms:" << std::endl;
568  first=false;
569  }
570 
571  std::cout << "\t" << group_name << ":" << n_hists_written + n_empty_hists_written
572  << " (no entries =" << n_empty_hists_written << ")" << std::endl;
573  }
574 
575  n_histograms+=hist_col.n();
576 
577  writer->upDir();
578  // if (!folder_name.empty()) {
579  // writer->upDir();
580  // }
581  }
582 
583 
584  UInt_t n_2d_histograms=0;
585  for (Histogram2DList_t::iterator col_iter=group._histogram2DList.begin();
586  col_iter!=group._histogram2DList.end();
587  col_iter++) {
588 
589  Histogram2DCollection_t &hist_col=*(*col_iter);
590  if (hist_col.n()==0) {
591  std::cerr << "HistMgr::writeHistograms> 2D Histogram collection " << col_iter.name()
592  << " is empty." << std::endl;
593  continue;
594  }
595 
596 
597  // the file is only created if the group contains at least one histogram;
598  if (!writer) {
599  writer=_writerKit->createWriter(group.header().fileName());
600  if (!folder_name.empty()) {
601  writer->enterDir(folder_name,true);
602  }
603  }
604 
605  writer->enterDir(col_iter.name(),true);
606 
607  UInt_t n_hists_written=0;
608  UInt_t n_empty_hists_written=0;
609  if (hist_col.is2D()) {
610  for (UInt_t major_i=0; major_i<(UInt_t) hist_col.nMajor(); major_i++) {
611  for (UInt_t hist_i=0; hist_i<(UInt_t) hist_col.nMinor(major_i); hist_i++) {
612  const FloatHistogram2D *a_hist=hist_col.histogram(major_i,hist_i);
613  if (a_hist->entries()>0) {
614  n_hists_written++;
615  }
616  else {
