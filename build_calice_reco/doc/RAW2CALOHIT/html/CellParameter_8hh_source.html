230  void adjustPedestalNoise(Float_t adc, UInt_t keep_n_minus_one) {
231  Float_t old_pedestal=getPedestal();
232  Float_t old_noise=getNoise();
233  setOldPedestal(old_pedestal);
234  setOldNoise(old_noise);
235 
236  // 1) first calculate sum of squares and sum from noise and pedestal
237  // 2) multiply sum of squares, and sum by (n-1)/n
238  // 3) add the new value and the new value squared
239  // 4) calculate new pedestal and new noise
240  Float_t inv_n=1./(keep_n_minus_one+1);
241  Float_t pedestal_sum=old_pedestal*keep_n_minus_one;
242  Float_t new_sum2=sqr(old_noise*(keep_n_minus_one))*inv_n + old_pedestal*pedestal_sum + adc*adc;
243  pedestal_sum+=adc;
244  Float_t new_pedestal = pedestal_sum * inv_n;
245  Float_t new_noise=sqrt((new_sum2-new_pedestal*pedestal_sum)/(keep_n_minus_one));
246 
247  setPedestal(new_pedestal);
248  setNoise(new_noise);
249  };
