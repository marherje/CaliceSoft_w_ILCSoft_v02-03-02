307  IMPL::ClusterImpl *a_cluster=new IMPL::ClusterImpl;
308  Double_t sum[3]={0.,0.,0.};

311  Double_t total_energy=0.;

313 
314  LinearRegression linreg[2];
315

317  hit_iter != cluster_iter->_hits.end();
318  hit_iter++) {
319  if ((*hit_iter)->getEnergy()>0.) {

321  }

323

325  hit_iter != cluster_iter->_hits.end();
326  hit_iter++) {
327

331  if (weight>0) {
332  for (UInt_t coord_i=0; coord_i<3; coord_i++) {
333  sum[coord_i]+=(*hit_iter)->getPosition()[coord_i] * weight;

366  hit_iter != cluster_iter->_hits.end();

397 #endif
