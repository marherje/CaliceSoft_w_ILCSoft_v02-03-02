103 void SimpleHcalCalibrationProcessor::processEvent(lcio::LCEvent *evt){
104  LCCollection *inCol = evt->getCollection(_inputColName);
105  if (inCol){
106  TriggerBits trigBits(evt->getParameters().getIntVal(PAR_TRIGGER_EVENT));
107  if (trigBits.isPurePedestalTrigger()) {
108  if (!_calcPed) resetPedestal();
109  sumupPedestal(inCol);
110  _pedEvents++;
111  }
112  if (trigBits.isBeamTrigger() && _calcPed) calcPedestal();
113 
114  if (_pedExists) {
115 #ifdef RECO_DEBUG
116  std::cout << "processing real beam event" << std::endl;
117 #endif
118 
119  LCCollectionVec *outCol = new LCCollectionVec(LCIO::RAWCALORIMETERHIT);
120 
121  for (unsigned i=0; i< static_cast<unsigned>(inCol->getNumberOfElements()); i++){
122  FastCaliceHit *hit = dynamic_cast<FastCaliceHit*>(inCol->getElementAt(i));
123  unsigned short mod = (hit->getModuleID() & 0xFF00) >> 8;
124  unsigned short cell = hit->getChip()*18 + hit->getChannel();
125 
126 #ifdef RECO_DEBUG
127  std::cout << "old hit: " << mod << " " << cell << " " << hit->getEnergyValue() << std::endl;
128 #endif
129  if ((mod > HCAL_N_MOD+1) || (cell > HCAL_N_CELL)) continue;
130  if (_stat[mod][cell] != 111) continue; //ignore cells with missing calibration
131  float ampl = hit->getEnergyValue() - _ped[mod][cell]; //ampl in adc counts
132 // float ampl = hit->getEnergyValue();
133  float amplErr = 0;
134  float N_pix = _Npix; //number of effective pixel
135  float xTalk = _xTalk; //crosstalk
136 
137  // correction factor
138  float corr_vas; // relative linearity correction factor
139  float corrErr; // uncertainty on relative correction
140  // correction with global factor on pixel level (involves gain and ic)
141 
142  //from satoru
143  if(_correction == "satoru")
144  {
145  ampl = N_pix*std::log(2 * xTalk / (xTalk - 1. + sqrt(xTalk*xTalk + 2 * xTalk * (1.-2.*ampl*_ic[mod][cell]/_gain[mod][cell] /N_pix) +1.))); //corrected ampl in pix
146 
147  ampl = ampl*_gain[mod][cell]/_ic[mod][cell]/_mip[mod][cell]; //ampl in mip
148  }
149 
150  //from vasiliy
151  else if(_correction == "vasiliy")
152  {
153  corr_vas = 1./(1.-_alpha*ampl*_ic[mod][cell]/_gain[mod][cell]);
154 
155  ampl /= _mip[mod][cell]; // amplitude in uncorrected MIP
156  ampl *= corr_vas;
157 
158  float cutAmpl = _pixCut*_gain[mod][cell]/_ic[mod][cell]/_mip[mod][cell];
159  if (ampl > cutAmpl) ampl = cutAmpl;
160  }
161 
162  else
163  ampl = ampl;
164 
165  corrErr = 0;
166 
167  if (ampl < 0.5) continue; // ignore hits<0.5MIP
168 
169  float mipErr = .05*_mip[mod][cell]; // relative mip uncertainty (global)
170  amplErr = ampl * sqrt(mipErr*mipErr + corrErr*corrErr);
171 
172  // create new CaliceHit and append to output collection
173  FastCaliceHit* newHit = new FastCaliceHit(hit->getModuleID(), hit->getChip(), hit->getChannel(),
174  ampl, amplErr, hit->getTimeStamp());
175 #ifdef RECO_DEBUG
176  std::cout << "new hit: " << mod << " " << cell << " " << ampl << std::endl;
177 #endif
178  outCol->addElement(newHit);
179  }
180  evt->addCollection(outCol,_outputColName);
181  }
182  }
