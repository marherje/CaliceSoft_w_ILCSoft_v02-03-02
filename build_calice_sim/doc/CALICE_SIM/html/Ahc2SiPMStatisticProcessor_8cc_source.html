273 
274  //--------------------------------------------------------------
275  //Get the list of collections in the event
276  const std::vector<std::string> *cnames = evt->getCollectionNames();
277 
278  for(unsigned int icol = 0; icol < _calorimInpCollections.size(); icol++)
279  {
280  if( std::find(cnames->begin(), cnames->end(), _calorimInpCollections.at(icol)) != cnames->end() )
281  {
282  string _inputColName = _calorimInpCollections.at(icol);
283 
284  LCCollection *inputCalorimCollection = NULL;
285  try{
286  inputCalorimCollection = evt->getCollection( _inputColName );
287  }
288  catch ( EVENT::DataNotAvailableException &e )

291  }
292 
293  LCCollection *noiseCol = NULL;
294  if(_doAddNoise)
295  {
296  try{
297  noiseCol = evt->getCollection( _noiseColName );
298  }
299  catch ( EVENT::DataNotAvailableException &e )
300  {
301  streamlog_out(WARNING)<<" Collection "<< _noiseColName <<" not available, skip this event"<<std::endl;
302  }
303  }
304 
305  if (inputCalorimCollection != 0){
306 
307  int nHits = inputCalorimCollection->getNumberOfElements();
308  streamlog_out(DEBUG) << "Number of hits: " << nHits << std::endl;
309 
310  /*the Mokka encoding string*/
311  _mokkaEncodingString = inputCalorimCollection->getParameters().getStringVal(LCIO::CellIDEncoding);
312 
313  /*clean up the container at the beginning of the event*/
314  delete _HitContainer;
315 
316  /*----------------------------------------------------------------------------------
317  First step: fill the container with hits E[MIPs]*/
318  this->fillContainer(inputCalorimCollection);
319 
320  /*---------------------------------------------------------------------------------
321  Next step: OPTICAL cross talk, still E[MIPs]
322  */
323  if(_doOpticalCrossTalk)
324  this->simulateOpticalCrossTalk();
325 
326  /*---------------------------------------------------------------------------------
327  Next step: Simulate SiPM behaviour,
328  i.e. convert the energy from MIPs to ADC scale
329  */
330  this->simulateSiPMBehaviour();
331

333  Next step: Add the noise from data, E[ADC]
334  */
335  if(_doAddNoise && noiseCol != 0) this->addNoise(noiseCol);
336 
337  /*---------------------------------------------------------------------------------
338  Next step: Finally, create the output collection, E[ADC]
339  */
340  createOutputCollection(evt, _calorimOutCollections.at(icol));
341 
342  streamlog_out(DEBUG) << " \n ---------> End Event: " << evtNumber << "!!! <-------------\n" << std::endl;
343 
344  _nEvt++;
345 
346  }//end if inputCol
347  }//end find collections
348  }//end for loop
349  }

358  void Ahc2SiPMStatisticProcessor::fillContainer(LCCollection *inputCol)
359  {
360  streamlog_out(DEBUG) << "\n\n Start to fill Hit container " << std::endl;
361 
362  /*create a new map container. Note the 'false' flag, which tells the container
363  that he is not responsible for the deletion of the contained elements.
364  We use this because at the end of the event we have all the elements from
365  the container into an LCCollection, and Marlin deletes them at the end of the event.*/
366 
367  _HitContainer = new MappedContainer<CalorimeterHitImpl>(_mapper, false);
368  _HitContainer->getDecoder()->setCellIDEncoding(_mokkaEncodingString);
369 
370  /*Copy the input collection into the container, which we'll modify afterwards*/
371  for (int iHit = 0; iHit < inputCol->getNumberOfElements(); ++iHit)
372  {
373  CalorimeterHit *hit = dynamic_cast<CalorimeterHit*>(inputCol->getElementAt(iHit));
374  int cellID0 = hit->getCellID0();
375 
376  CalorimeterHitImpl *newHit = new CalorimeterHitImpl();
377  newHit->setTime(hit->getTime());
378  newHit->setEnergy(hit->getEnergy());
379  newHit->setCellID0(cellID0);
380  newHit->setPosition(hit->getPosition());
381 
382  streamlog_out(DEBUG) << "\n---------------" << std::endl;
383  streamlog_out(DEBUG) << " currentHit with cell ID " << cellID0 <<", energyInMIPs "<< hit->getEnergy()
384  << ", I/J/K " << _HitContainer->getDecoder()->getIFromCellID(cellID0)
385  << "/" << _HitContainer->getDecoder()->getJFromCellID(cellID0)
386  << "/" << _HitContainer->getDecoder()->getKFromCellID(cellID0)
387  << std::endl;
388 
389  _HitContainer->fillByCellID(cellID0, newHit);
390  }
391 
392  streamlog_out(DEBUG) <<"\n After: HitContainer has "
393  <<_HitContainer->getAllElements().size()
394  <<" elements"<<std::endl;
395 
396  }
397 
398  /*********************************************************************************/
399  /* */
400  /* Note: during cross-talk, 10% of the energy leaks to the neighbouring */
401  /* tiles, but the leaked energy is NOT subtracted from the energy because */
402  /* the measured MIP is actually only 90%MIPs. For more details, see page 101 in */
403  /* http://www-library.desy.de/preparch/desy/thesis/desy-thesis-10-006.pdf */
404  /* */
405  /*********************************************************************************/
406  void Ahc2SiPMStatisticProcessor::simulateOpticalCrossTalk()
407  {
408  streamlog_out(WARNING)<< "\n\n Start to simulate optical cross-talk";
409 
410  std::vector<CalorimeterHitImpl*> HitVec = _HitContainer->getAllElements();
411  streamlog_out(WARNING)<< " for " << HitVec.size() << " hits" << std::endl;
412 
413  /*counter for debug purposes*/
414  int countNewCells = 0;
415  int countGanged = 0;
416 
417  // Generate a new vector to save the original hits information
418  // Loop over the original hits, and add optical xtalk into the
419  // neighbours cell.
420  // It is only first order optical xtalk has been applied here.
421  std::vector<CalorimeterHitImpl> originalHitVec;
422  for (std::vector<CalorimeterHitImpl *>::iterator iter = HitVec.begin();
423  iter != HitVec.end(); ++iter)
424  {
425  CalorimeterHitImpl *currentHit = (*iter);
426  originalHitVec.push_back((*currentHit));
427  }
428 
429  /*------------------------------------------------------------------
430  Start to loop over original hits
431  ------------------------------------------------------------------*/
432  for (std::vector<CalorimeterHitImpl>::iterator iter = originalHitVec.begin();
433  iter != originalHitVec.end(); ++iter)
434  {
435  /*set the encoding string back to the Mokka encoding string*/
436  _HitContainer->getDecoder()->setCellIDEncoding(_mokkaEncodingString);
437 
438  int cellID = (*iter).getCellID0();
439  CalorimeterHitImpl *currentHit = &(*iter);
440 
441  _cellDescriptions->getDecoder()->setCellIDEncoding(_mokkaEncodingString);
442 
443  CellDescription* currentCellDescription = _cellDescriptions->getByCellID(cellID);
444  const float currentTileSize = currentCellDescription->getSizeX();
445 
446  streamlog_out(WARNING)<<"------------"<<endl;
447  streamlog_out(WARNING)<<"i="<<countGanged<<" hit I/J/K: "<< _HitContainer->getDecoder()->getIFromCellID(cellID)
448  <<"/"<<_HitContainer->getDecoder()->getJFromCellID(cellID)
449  <<"/"<<_HitContainer->getDecoder()->getKFromCellID(cellID)
450  <<" currentHit: "<<currentHit
451  <<" cellID: "<<cellID
452  <<" energy[MIPs]: "<<currentHit->getEnergy()
453  <<" currentTileSize: "<<currentTileSize
454  <<endl;
455  countGanged ++;
456 
457  //Get the light leakage value for the hit in the layer
458  float _lightLeakageValue = 0.0;
459 
460  if(m_leakage.find(_HitContainer->getDecoder()->getKFromCellID(cellID)) != m_leakage.end())
461  _lightLeakageValue = m_leakage[_HitContainer->getDecoder()->getKFromCellID(cellID)];
462  else
463  continue;
464 
465  /*start loop over neighbours*/
466  const std::vector<int> neighboursVec = _cellNeighbours->getByCellID(cellID)->getNeighbours(CellNeighbours::direct, CellNeighbours::module);
467  streamlog_out(WARNING)<<" ->number of neighbours found: "<<neighboursVec.size()<<endl;
468 
469  /*before getting the neighbours, set the encoding string to the string used to
470  set the neighbours*/
471  const std::string neighboursEncodingString = CALICE::CellNeighboursProcessor::getEncodingString(_cellNeighboursProcessorName);
472  streamlog_out(WARNING)<<" neighboursEncodingString: "<<neighboursEncodingString<<endl;
473 
474  /*---------------------------------------------------------------
475  loop over neighbours
476  -----------------------------------------------------------------*/
477  for (unsigned int i = 0; i < neighboursVec.size(); ++i)
478  {
479  streamlog_out(WARNING)<<" \n i="<<i<<endl;
480  _cellDescriptions->getDecoder()->setCellIDEncoding(neighboursEncodingString);
481  streamlog_out(WARNING)<<" neighboursEncodingString: "<<neighboursEncodingString<<endl;
482 
483  streamlog_out(WARNING)<<" neighbours: I/J/K="
484  <<_cellDescriptions->getDecoder()->getIFromCellID(neighboursVec[i])
485  <<"/"<<_cellDescriptions->getDecoder()->getJFromCellID(neighboursVec[i])
486  <<"/"<<_cellDescriptions->getDecoder()->getKFromCellID(neighboursVec[i])
487  <<" endcoding string: "<<_cellDescriptions->getDecoder()->getCellIDEncoding()
488  <<endl;
489 
490  CellDescription *neighbourCellDescription = _cellDescriptions->getByCellID(neighboursVec[i]);
491  if (neighbourCellDescription == NULL)
492  {
493  streamlog_out(WARNING)<<" sorry, no cell description found for neighbour "<<i<<endl;
494  continue;
495  }
496 
497  const float neighbourTileSize = neighbourCellDescription->getSizeX();
498  streamlog_out(WARNING)<<" neighbourTileSize: "<<neighbourTileSize<<endl;
499 
500  /*Define fraction of leakage going into neighbouring cells;
501  for example, if the current cell has neighbours of the same size,
502  and if the total leakage is 10%, each neighbour will get
503  an energy fraction of (leakageFraction * totalLeakage) = 0.25 * 0.10 = 0.025*/
504  float leakageFraction = 1.0/4.0;
505 
506  /*if the neighbours are larger or equal to the current tile,
507  they will get a quarter (1/4) of the leaked energy of the current tile;
508  else, the leakageFraction has to consider the smaller size of the neighbours,
509  since for the leakage is equally distributed at each side of the current tile*/
510  if (neighbourTileSize >= currentTileSize) leakageFraction *= 1;
511  else if (neighbourTileSize < currentTileSize) leakageFraction *= neighbourTileSize/currentTileSize;
512 
513  /*now we can calculate the energy which leaks from the current tile to the neighbours;
514  the factor A=(1. - _lightLeakage)
515  plays the role of an effective MIP/GeV factor. We do this because, for the muon case,
516  if we sum up the energy, we need to get the 1 MIP.
517  */
518  float leakedEnergy = leakageFraction * _lightLeakageValue * (currentHit->getEnergy()) / (1. - _lightLeakageValue);
519  float time = currentHit->getTime();
520 
521  streamlog_out(WARNING)<<" lightLeakage: "<<_lightLeakageValue<<" energy: "<<currentHit->getEnergy()<<std::endl;
522  streamlog_out(WARNING)<<" leakageFraction: "<<leakageFraction<<" leakedEnergy: "<<leakedEnergy<<endl;
523  streamlog_out(WARNING)<<" looking in list for I/J/K="
524  <<_cellDescriptions->getDecoder()->getIFromCellID(neighboursVec[i])
525  <<"/"<<_cellDescriptions->getDecoder()->getJFromCellID(neighboursVec[i])
526  <<"/"<<_cellDescriptions->getDecoder()->getKFromCellID(neighboursVec[i])
527  <<endl;
528 
529  _HitContainer->getDecoder()->setCellIDEncoding(neighboursEncodingString);
530  CalorimeterHitImpl *neighbourHit = _HitContainer->getByCellID(neighboursVec[i]);
531 
532  if (neighbourHit == NULL)
533  {
534  neighbourHit = new CalorimeterHitImpl();
535  neighbourHit->setEnergy(leakedEnergy);
536  neighbourHit->setTime(time);
537 
538  countNewCells++;
539  streamlog_out(WARNING)<<" new hit, leaked energy in the neighbour: "<<neighbourHit->getEnergy()<<std::endl;
540 
541  /*need to get the cell ID in the Mokka encoding, to be able to fill the gangedContainer*/
542  const int I = _HitContainer->getDecoder()->getIFromCellID(neighboursVec[i]);
543  const int J = _HitContainer->getDecoder()->getJFromCellID(neighboursVec[i]);
544  const int K = _HitContainer->getDecoder()->getKFromCellID(neighboursVec[i]);
545 
546  _HitContainer->getDecoder()->setCellIDEncoding(_mokkaEncodingString);
547  const int cellIDinMokkaEncoding = _HitContainer->getDecoder()->getCellID(I, J, K);
548  neighbourHit->setCellID0(cellIDinMokkaEncoding);
549  _HitContainer->fillByCellID(cellIDinMokkaEncoding, neighbourHit);
550  }
551  else
552  {
553  streamlog_out(WARNING)<<" Guess what, hit alread exists, energy="<<neighbourHit->getEnergy()<<endl;
554 
555  /*add up the leakedEnergy to the original energy of the neighbours*/
556  neighbourHit->setEnergy(neighbourHit->getEnergy() + leakedEnergy);
557  }
558 
559  streamlog_out(WARNING)<<" final energy in the neighbour: "<<neighbourHit->getEnergy()<<std::endl;
560  }/*---------------- end loop over neighbours vector -----------------------*/
561 
562  }/*------------------- end loop over ganged iterator ----------------------------*/
563 
564 
565  /*************************************************************/
566  streamlog_out(WARNING)<<"\n After cross-talk: HitContainer has "
567  <<_HitContainer->getAllElements().size()
568  <<"elements, from which "<<countNewCells<<" new hits"
569  <<endl;
570 
571  }

580  void Ahc2SiPMStatisticProcessor::simulateSiPMBehaviour()
581  {
582  streamlog_out(DEBUG) << " Start to simulate SiPM behaviour " <<std::endl;
583 
584  /*now the hit container contains all hits/subhits */
585  std::vector<CalorimeterHitImpl*> HitVec = _HitContainer->getAllElements();
586  for (std::vector<CalorimeterHitImpl*>::iterator iter = HitVec.begin(); iter != HitVec.end(); ++iter)
587  {
588  int cellID = (*iter)->getCellID0();
589  CalorimeterHitImpl *currentHit = (*iter);
590  float simHitMIP = currentHit->getEnergy();
591 
592  _HitContainer->getDecoder()->setCellIDEncoding(_mokkaEncodingString);
593 
594  streamlog_out(DEBUG) << "\n---------------" << std::endl;
595  streamlog_out(DEBUG) << " currentHit with cell ID " << cellID <<", energyInMIPs "<< simHitMIP
596  << ", I/J/K " << _HitContainer->getDecoder()->getIFromCellID(cellID)
597  << "/" << _HitContainer->getDecoder()->getJFromCellID(cellID)
598  << "/" << _HitContainer->getDecoder()->getKFromCellID(cellID)
599  << std::endl;
600 
601  streamlog_out(DEBUG) <<" calibrationEncodingString: "<< _calibContainer->getDecoder()->getCellIDEncoding() << std::endl;
602 
603  //Get calibrations for the cell
604  Ahc2Calibrations *calibration = _calibContainer->getByCellID(cellID);
605 
606  if (calibration == NULL)
607  {
608  streamlog_out(DEBUG)<<" Sorry, no calibration found for cellID " << cellID <<std::endl;
609  _HitContainer->deleteByCellID(cellID);
610  continue;
611  }
612 
613  //Get Status of cell (if dead or default values)
614  const Ahc2CalibrationStatusBits bits = Ahc2CalibrationStatusBits( calibration->getStatus() );
615 
616  streamlog_out(DEBUG) << " STATUS bits: " << bits << std::endl;
617 
618  if ( _filterDeadCells && bits.isDead() )
619  {
620  streamlog_out(DEBUG) << " SKIP -- channel is marked as dead: " << std::endl;
621  _HitContainer->deleteByCellID(cellID);

645  //SimpleValue *temperatureObject = calibration->getTemperature();
646  //const float temperature = temperatureObject->getValue();

711  // saturate
712  // smear
713  // unsaturate
714  // energy pixels to mip
715

730 
731  if(_doAdditionalSmearing) {
732  satSmearedHitPx = satSmearedHitPx + _randomGenerator->Gaus(0., _pixSpread); //smearing due to pixel spread
733  }
734 
735  if(satSmearedHitPx > 0)
736  {
737  //if physics mode divide the gain in calib mode by Intercalibration Physics/Calib
738  if(_physicsMode)
739  simHitADC = satSmearedHitPx * gainValue / interCalibrationValuePhys; //energy pixels -> ADC
740  else
741  simHitADC = satSmearedHitPx * gainValue;
742  }
743  else
744  simHitADC = 0;
745

747  streamlog_out(DEBUG) << "- prob: " << prob << std::endl;
748  streamlog_out(DEBUG) << "- satSmearedHitPx: " << satSmearedHitPx << std::endl;
749  //streamlog_out(DEBUG) << "- unsatSmearedHitPx: " << unsatSmearedHitPx << std::endl;
750  streamlog_out(DEBUG) << "- simHitADC: " << simHitADC << std::endl;
751 
752  }
753  else
754  {
755  if(_physicsMode)
756  simHitADC = satHitPx * gainValue / interCalibrationValuePhys;
757  else
758  simHitADC = satHitPx * gainValue;
759  }
760  }
761  else
762  {
763  if(_physicsMode)
764  simHitADC = simHitPx * gainValue / interCalibrationValuePhys;
765  else
766  simHitADC = simHitPx * gainValue;
767  }

770  simHitADC = interCalibrationValue * MAXADC;
771 
772  streamlog_out(DEBUG) << "- simHitADC (after saturation & smearing): " << simHitADC << std::endl;
773 
774  /* now we store the ADC channels per channel*/
775  currentHit->setEnergy(simHitADC);
776 
777  delete satFunc;
778 
779  }/*------------------------ end loop over gangedVector -----------------------------*/
