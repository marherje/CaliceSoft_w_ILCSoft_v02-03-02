144  void TempCalHit::addContribution(long long id, double energy, double time) {
145  if( _simHits->count(id)>0 ) {
146  // contribution already exists
147  HitContrib& contr = _simHits->find(id)->second;
148  contr.increment(energy,time);
149  }
150  else {
151  // new contribution
152  HitContrib contr(energy,time);
153  _simHits->insert( pair<long long, HitContrib>(id, contr) );
154  }
155  _energy += energy;
156  if( time < _time ) _time = time;
157  }
