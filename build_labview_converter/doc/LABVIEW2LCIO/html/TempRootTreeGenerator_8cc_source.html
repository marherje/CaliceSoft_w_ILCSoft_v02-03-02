The May and June 2018 data require additional mapping files because here, the temperature was read out in the order of the LDA port number which is not equal to the layer number. 5 LDA Port 0 T1 T2 T3 T4 T5 T6 TDIF TPWR

7 LDA Port 2 T1 T2 T3 T4 T5 T6 TDIF TPWR

130  //Timestamp= col->getParameters().getIntVal("Timestamp_i");//TimeStamp();//airqui

132  int countPort = 0;

135  int nLayers=0;
136 
137  //read mapping files
138  ifstream inputFile;

140  for(int i = 0; i < size; ++i){
141  inputFile >> mapping[i];
142  //compute number of layers in detector

149  for(int ielm = 0; ielm < col->getNumberOfElements(); ielm++)
150  {

154  float t1=lBlock.GetT1()/10.;

156  float t3=lBlock.GetT3()/10.;
157  float t4=lBlock.GetT4()/10.;
158  float t5=lBlock.GetT5()/10.;
159  float t6=lBlock.GetT6()/10.;
160  float tdif=lBlock.GetTDIF();
161  float tpwr=lBlock.GetTPWR();
162 
163  streamlog_out(MESSAGE) << "Temperatures : " << t1 << " " << t2 << " " << t3 << " " << t4 << " " << t5 << " " << t6 << " " << tdif << " " << tpwr << endl;
164 
165  //apply the mapping
166  _hFill.T1[mapping[countPort]-1] = t1;
167  _hFill.T2[mapping[countPort]-1] = t2;
168  _hFill.T3[mapping[countPort]-1] = t3;
169  _hFill.T4[mapping[countPort]-1] = t4;

173  _hFill.TPWR[mapping[countPort]-1] = tpwr;

175  //keep track of the current port number

177  countPort++;
178  }
179  else{
180  countPort=0;
181  }
182  }
183 
184  _hFill.nLayers = countPort;
